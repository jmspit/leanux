97  udevMode modeFromMajorMinor( const MajorMinor& m, const std::string &udev_path ) {
98  std::stringstream ss;
99  ss << "b" << m;
100  if ( util::fileReadAccess( udev_path + ss.str() ) ) return umBMM; else {
101  ss.str("");
102  ss << "block" << m;
103  if ( util::fileReadAccess( udev_path + ss.str() ) ) return umBlockMM;
104  else {
105  ss.str("");
106  ss << "block:" << MajorMinor::getNameByMajorMinor( m );
107  if ( util::fileReadAccess( udev_path + ss.str() ) ) return umBlockName; else {
108  ss.str("");
109  ss << "unable to find udev block device info in path ";
110  ss << udev_path << " for maj:min " << m;
111  throw Oops( __FILE__, __LINE__, ss.str() );
112  }
113  }
114  }
115  return umBlockMM;
116  }

118  void init() {
119  //detect udev path
120  if ( util::directoryExists( "/dev/.udev/db" ) ) {
121  udev_path = "/dev/.udev/db/";
122  } else if ( util::directoryExists( "/run/udev/data" ) ) {
123  udev_path = "/run/udev/data/";
124  } else if ( util::directoryExists( "/var/run/udev/data" ) ) {
125  udev_path = "/var/run/udev/data/";
126  }
127  if ( udev_path == "" ) throw Oops( __FILE__, __LINE__, "cannot find udev run data" );
128  //detect udev block device format
129  std::stringstream ss;
130  //first attempt to stat the root (/) filesystem to get a disk
131  struct stat rootfs;
132  if ( !stat("/",&rootfs) ) {
133  MajorMinor rootdevice(rootfs.st_dev);
134  // btrfs (sigh...) has no sane 'major', so btrfs breaks the stat call returning major 0.
135  if ( rootdevice.isValid() && rootdevice.getMajor() != 0 ) {
136  udev_mode = modeFromMajorMinor( rootdevice, udev_path );
137  return;
138  }
139  }
140  std::list<MajorMinor> wd;
141  enumWholeDisks( wd );
142  if ( wd.size() > 0 ) {
143  udev_mode = modeFromMajorMinor( wd.front(), udev_path );
144  } else throw Oops( __FILE__, __LINE__, "did not detect any whole disks" );
145  }

147  MajorMinor& MajorMinor::operator=(const std::string &s ) {
148  std::string smajor = "";
149  std::string sminor = "";
150  int state = 0;
151  size_t p = 0;
152  while ( p < s.length() ) {
153  switch( state ) {
154  case 0:
155  if ( s[p] == ':' ) state = 1; else
156  if ( !isdigit( s[p] ) ) throw Oops( __FILE__, __LINE__, "invalid MajorMinor std::string" );
157  else smajor += s[p];
158  p++;
159  break;
160  case 1:
161  if ( !isdigit( s[p] ) ) throw Oops( __FILE__, __LINE__, "invalid MajorMinor std::string" );
162  else sminor += s[p];
163  p++;
164  break;
165  }
166  }
167  dev_ = MKDEV( strtoul( smajor.c_str(), 0, 10 ), strtoul( sminor.c_str(), 0, 10 ) );
168  return *this;
169  }

171  void MajorMinor::buildCache() {
172  name2mm_.clear();
173  mm2name_.clear();
174  std::ifstream i( "/proc/diskstats" );
175  if ( !i.good() ) throw Oops( __FILE__, __LINE__, "failed to open '/proc/diskstats'" );
176  unsigned int major, minor;
177  std::string device;
178  while ( i.good() ) {
179  i >> major >> minor >> device;
180  name2mm_[ device ] = MajorMinor( major, minor );
181  mm2name_[ MajorMinor( major, minor ) ] = device;
182  //check devicefile
183  std::list<std::string> aliasses;
184  MajorMinor( major, minor ).getAliases( aliasses );
185  bool nodevice = true;
186  for ( std::list<std::string>::const_iterator a = aliasses.begin(); a != aliasses.end(); a++ ) {
187  std::string rp = util::realPath( *a );
188  if ( rp != "" ) {
189  file2mm_[ rp ] = MajorMinor( major, minor );
190  mm2file_[ MajorMinor( major, minor ) ] = rp;
191  nodevice = false;
192  break;
193  }
194  }
195  if ( nodevice ) {
196  std::string rp = "/dev/" + device;
197  file2mm_[ rp ] = MajorMinor( major, minor );
198  mm2file_[ MajorMinor( major, minor ) ] = rp;
199  }
200 
201  i.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
202  }
203  }

321  MajorMinor MajorMinor::deriveWholeDisk( const MajorMinor& partition ) {
322  if ( isSCSIDisk( MAJOR(partition.dev_) ) ) {
323  return MajorMinor( MKDEV(MAJOR(partition.dev_), MINOR(partition.dev_) - MINOR(partition.dev_) % 16 ) );
324  } else if ( isIDEDisk( MAJOR(partition.dev_) ) ) {
325  return MajorMinor( MKDEV(MAJOR(partition.dev_), MINOR(partition.dev_) - MINOR(partition.dev_) % 64 ) );
326  } else if ( isVirtIODisk( MajorMinor( MAJOR(partition.dev_), 0 ) ) ) {
327  return MajorMinor( MKDEV(MAJOR(partition.dev_), MINOR(partition.dev_) - MINOR(partition.dev_) % 16 ) );
328  } else if ( isNVMeDisk( MajorMinor( MAJOR(partition.dev_), 0 ) ) ) {
329  std::string devname = getNameByMajorMinor( partition );
330  unsigned int host = 0;
331  unsigned int disk = 0;
332  int r = sscanf( devname.c_str(), "nvme%un%u", &host, &disk );
333  if ( r != 2 ) throw leanux::Oops( __FILE__, __LINE__, "unable to parse NVMe device name" );
334  std::stringstream ss;
335  ss << std::fixed << "nvme" << host << "n" << disk;
336  return getMajorMinorByName( ss.str() );
337  } else if ( isMMCDisk( partition ) ) {
338  std::string devname = getNameByMajorMinor( partition );
339  unsigned int disk = 0;
340  int r = sscanf( devname.c_str(), "mmcblk%u", &disk );
341  if ( r != 1 ) throw leanux::Oops( __FILE__, __LINE__, "unable to parse MMC device name" );
342  std::stringstream ss;
343  ss << std::fixed << "mmcblk" << disk;
344  return getMajorMinorByName( ss.str() );
345  } else if ( isBCacheDisk( partition ) ) {
346  std::string devname = getNameByMajorMinor( partition );
347  unsigned int disk = 0;
348  int r = sscanf( devname.c_str(), "bcache%u", &disk );
349  if ( r != 1 ) throw leanux::Oops( __FILE__, __LINE__, "unable to parse bcache device name" );
350  std::stringstream ss;
351  ss << std::fixed << "bcache" << disk;
352  return getMajorMinorByName( ss.str() );
353  }
354  return partition;
355  }

385  DeviceClass MajorMinor::getClass() const {
386  if ( MajorMinor::isSCSIDisk( getMajor() ) ) {
387  if ( isPartition() ) return SCSIDiskPartition; else return SCSIDisk;
388  } else if ( MajorMinor::isIDEDisk( getMajor() ) ) {
389  if ( isPartition() ) return IDEDiskPartition; else return IDEDisk;
390  } else if ( MajorMinor::isVirtIODisk( *this ) ) {
391  if ( isPartition() ) return VirtioDiskPartition; else return VirtioDisk;
392  } else if ( MajorMinor::isNVMeDisk( *this ) ) {
393  if ( isPartition() ) return NVMeDiskPartition; else return NVMeDisk;
394  } else if ( MajorMinor::isMMCDisk( *this ) ) {
395  if ( isPartition() ) return MMCDiskPartition; else return MMCDisk;
396  } else if ( MajorMinor::isBCacheDisk( *this ) ) {
397  if ( isPartition() ) return BCacheDiskPartition; else return BCacheDisk;
398  } else {
399  std::string lvname;
400  std::string dm_uuid;
401  lvname = getLVName();
402  if ( lvname != "" ) return LVM;
403  dm_uuid = getDMUUID();
404  if ( dm_uuid.find("mpath-") != std::string::npos ) return MultiPath;
405 
406  switch ( getMajor() ) {
407  case 1 : return RAMDisk;
408  case 2 : return FloppyDisk;
409  case 7 : return Loopback;
410  case 9 : return MetaDisk;
411  case 11 : return SCSICD;
412  default : return Unknown;
413  }
414  }
415  return Unknown;
416  }
