70  void RealtimeSampler::sampleXProcView() {
71  xprocview_.t1 = xprocview_.t2;
72  xprocview_.pidargs.clear();
73  xprocview_.piduids.clear();
74  if ( !xprocview_.disabled ) {
75  procsnap1_ = procsnap2_;
76  gettimeofday( &xprocview_.t2, 0 );
77  double dt = util::deltaTime( xprocview_.t1, xprocview_.t2 );
78  util::Stopwatch sw;
79  process::getAllProcPidStat( procsnap2_ );
80  double duration = sw.stop();
81  if ( duration > 29.22 ) {
82  xprocview_.disabled = true;
83  system::getNumProcesses( &disabled_procs_ );
84  } else {
85  deltaProcPidStats( procsnap1_, procsnap2_, xprocview_.delta );
86  // @todo dirty hack to solve a nasty kernel reporting problem - some values are way too high
87  // https://bugzilla.kernel.org/show_bug.cgi?id=198245
88  for ( process::ProcPidStatDeltaVector::iterator i = xprocview_.delta.begin(); i != xprocview_.delta.end(); i++ ) {
89  if ( (*i).delayacct_blkio_ticks / dt > 1 ) (*i).delayacct_blkio_ticks = 0;
90  }
91  // @todo dirty hack to solve a nasty kernel reporting problem - some values are way too high
92  process::StatsSorter top_sorter( process::StatsSorter::top );
93  sort( xprocview_.delta.begin(), xprocview_.delta.end(), top_sorter );
94  for ( process::ProcPidStatDeltaVector::iterator i = xprocview_.delta.begin(); i != xprocview_.delta.end(); i++ ) {
95  (*i).delayacct_blkio_ticks /= dt;
96  (*i).majflt /= dt;
97  (*i).minflt /= dt;
98  (*i).stime /= dt;
99  (*i).utime /= dt;
100  uid_t uid;
101  process::getProcUid( (*i).pid, uid );
102  xprocview_.piduids[(*i).pid] = uid;
103  xprocview_.pidargs[(*i).pid] = process::getProcCmdLine( (*i).pid );
104  }
105  }
106  } else {
107  unsigned long procs_now;
108  system::getNumProcesses( &procs_now );
109  if ( (double)procs_now < (double)disabled_procs_ * 0.9 ) {
110  disabled_procs_ = 0;
111  xprocview_.disabled = false;
112  }
113  }
114  xprocview_.sample_count++;
115  }

133  xnetview_.sample_count++;

136  void RealtimeSampler::sampleXIOView() {
137  block::DeviceStatsMap devicestats;
138  block::MajorMinorVector sorted;
139  diskstats1_ = diskstats2_;
140  xioview_.t1 = xioview_.t2;
141  gettimeofday( &xioview_.t2, 0 );
142  double dt = util::deltaTime( xioview_.t1, xioview_.t2 );
143  block::getStats( diskstats2_ );
144  xioview_.iosorted.clear();
145  xioview_.iostats.clear();
146  xioview_.mountsorted.clear();
147  xioview_.mountstats.clear();
148  block::deltaDeviceStats( diskstats1_, diskstats2_, devicestats, sorted );
149  block::StatsSorter sorter(&devicestats);
150  sort( sorted.begin(), sorted.end(), sorter );
151 
152  for ( block::MajorMinorVector::const_iterator s = sorted.begin(); s != sorted.end(); s++ ) {
153  XIORec rec;
154  rec.device = (*s).getName();
155  rec.util = devicestats[*s].io_ms / 1000.0 / dt;
156  rec.rs = devicestats[*s].reads / dt;
157  rec.ws = devicestats[*s].writes / dt;
158  rec.rbs = devicestats[*s].read_sectors * (*s).getSectorSize() / dt;
159  rec.wbs = devicestats[*s].write_sectors * (*s).getSectorSize() / dt;
160  if ( devicestats[*s].reads != 0 )
161  rec.artm = devicestats[*s].read_ms / 1000.0 / devicestats[*s].reads;
162  else
163  rec.artm = 0;
164  if ( devicestats[*s].writes != 0 )
165  rec.awtm = devicestats[*s].write_ms / 1000.0 / devicestats[*s].writes;
166  else
167  rec.awtm = 0;
168  if ( (devicestats[*s].reads+devicestats[*s].writes) != 0 )
169  rec.svctm = devicestats[*s].io_ms / 1000.0 / (devicestats[*s].reads+devicestats[*s].writes);
170  else
171  rec.svctm = 0;
172  if ( rec.svctm != 0 )
173  rec.qsz = (rec.awtm+rec.awtm)/rec.svctm;
174  else
175  rec.qsz = 0;
176  rec.iodone_cnt = devicestats[*s].iodone_cnt / dt;
177  rec.iorequest_cnt = devicestats[*s].iorequest_cnt / dt;
178  rec.ioerr_cnt = devicestats[*s].ioerr_cnt / dt;
179  if ( xioview_.iostats.find( (*s).getName() ) == xioview_.iostats.end() ) {
180  xioview_.iostats[ (*s).getName() ] = rec;
181  if ( (*s).isWholeDisk() ) {
182  xioview_.iosorted.push_back((*s).getName());
183  }
184  }
185  }
186  vmem::getSwapInfo( swaps_ );
187  xioview_.fsbytes1 = xioview_.fsbytes2;
188  std::map<block::MajorMinor,block::MountInfo> mounts;
189  enumMounts( mounts, devicefilecache_ );
190 
191 
192  for ( std::map<block::MajorMinor,block::MountInfo>::const_iterator m = mounts.begin(); m != mounts.end(); ++m ) {
193  xioview_.fsbytes2[m->second.mountpoint] = block::getMountUsedBytes( m->second.mountpoint );
194  }
195  // map MajorMinor to mountpoint
196  for ( block::MajorMinorVector::const_iterator s = sorted.begin(); s != sorted.end(); s++ ) {
197  XIORec rec;
198  std::map<block::MajorMinor,block::MountInfo>::const_iterator m = mounts.find(*s);
199  if ( m != mounts.end() ) {
200  xioview_.mountstats[m->second.mountpoint].util = xioview_.iostats[(*s).getName()].util;
201  xioview_.mountstats[m->second.mountpoint].rs = xioview_.iostats[(*s).getName()].rs;
202  xioview_.mountstats[m->second.mountpoint].ws = xioview_.iostats[(*s).getName()].ws;
203  xioview_.mountstats[m->second.mountpoint].rbs = xioview_.iostats[(*s).getName()].rbs;
204  xioview_.mountstats[m->second.mountpoint].wbs = xioview_.iostats[(*s).getName()].wbs;
205  xioview_.mountstats[m->second.mountpoint].artm = xioview_.iostats[(*s).getName()].artm;
206  xioview_.mountstats[m->second.mountpoint].awtm = xioview_.iostats[(*s).getName()].awtm;
207  xioview_.mountstats[m->second.mountpoint].svctm = xioview_.iostats[(*s).getName()].svctm;
208  xioview_.mountstats[m->second.mountpoint].svctm = xioview_.iostats[(*s).getName()].svctm;
209  xioview_.mountstats[m->second.mountpoint].device = m->second.mountpoint;
210  xioview_.mountstats[m->second.mountpoint].growths = ((double)xioview_.fsbytes2[m->second.mountpoint] - (double)xioview_.fsbytes1[m->second.mountpoint])/dt;
211  xioview_.mountsorted.push_back(m->second.mountpoint);
212  } else {
213  //include swap filesystems
214  for ( std::list<vmem::SwapInfo>::const_iterator i = swaps_.begin(); i != swaps_.end(); i++ ) {
215  if ( block::getFileMajorMinor( (*i).devicefile ) == *s ) {
216  block::MountInfo mi;
217  mi.attrs = "";
218  mi.device = "";
219  mi.fstype = "swap";
220  mi.mountpoint = "swap:" + (*i).devicefile;
221  xioview_.mountstats[mi.mountpoint].util = xioview_.iostats[(*s).getName()].util;
222  xioview_.mountstats[mi.mountpoint].rs = xioview_.iostats[(*s).getName()].rs;
223  xioview_.mountstats[mi.mountpoint].ws = xioview_.iostats[(*s).getName()].ws;
224  xioview_.mountstats[mi.mountpoint].rbs = xioview_.iostats[(*s).getName()].rbs;
225  xioview_.mountstats[mi.mountpoint].wbs = xioview_.iostats[(*s).getName()].wbs;
226  xioview_.mountstats[mi.mountpoint].artm = xioview_.iostats[(*s).getName()].artm;
227  xioview_.mountstats[mi.mountpoint].awtm = xioview_.iostats[(*s).getName()].awtm;
228  xioview_.mountstats[mi.mountpoint].svctm = xioview_.iostats[(*s).getName()].svctm;
229  xioview_.mountstats[mi.mountpoint].svctm = xioview_.iostats[(*s).getName()].svctm;
230  xioview_.mountstats[mi.mountpoint].device = mi.mountpoint;
231  xioview_.mountsorted.push_back(mi.mountpoint);
232  break;
233  }
234  }
235  }
236  }
237  xioview_.sample_count++;
238  }

240  void RealtimeSampler::sampleXSysView( int cpubarheight ) {
241  cpustat1_ = cpustat2_;
242  sched1_ = sched2_;
243  vmstat1_ = vmstat2_;
244  xsysview_.t1 = xsysview_.t2;
245  gettimeofday( &xsysview_.t2, 0 );
246  cpu::getCPUTopology( xsysview_.cpu_topo );
247  cpu::getCPUStats( cpustat2_ );
248  sched2_ = cpu::getSchedInfo();
249  vmem::getVMStat( vmstat2_ );
250  vmem::getSwapInfo( swaps_ );
251  mounted_bytes_1_ = mounted_bytes_2_;
252  mounted_bytes_2_ = leanux::block::getMountUsedBytes();
253  xsysview_.cpu_delta.clear();
254  leanux::cpu::deltaStats( cpustat1_, cpustat2_, xsysview_.cpu_delta );
255  //normalize the cpu_delta to sample interval resulting in CPU seconds/clock second.
256  double dt = util::deltaTime( xsysview_.t1, xsysview_.t2 );
257  for ( cpu::CPUStatsMap::iterator c = xsysview_.cpu_delta.begin(); c != xsysview_.cpu_delta.end(); ++c ) {
258  cpu::normalizeCPUstats( c->second, dt, 1 );
259  }
260  cpu::getCPUTotal( xsysview_.cpu_delta, xsysview_.cpu_total );
261 
262  if ( xsysview_.sample_count > 0 ) {
263 
264  xsysview_.cpu_seconds = xsysview_.cpu_total.steal +
265  xsysview_.cpu_total.softirq +
266  xsysview_.cpu_total.irq +
267  xsysview_.cpu_total.iowait +
268  xsysview_.cpu_total.system +
269  xsysview_.cpu_total.nice +
270  xsysview_.cpu_total.user;
271 
272  xsysview_.time_slice = xsysview_.cpu_seconds / (double)(sched2_.ctxt - sched1_.ctxt );
273  xsysview_.ctxsws = (double)(sched2_.ctxt - sched1_.ctxt )/ dt;
274  xsysview_.forks = (double)(sched2_.processes - sched1_.processes )/ dt;
275 
276  cpu::getLoadAvg( xsysview_.loadavg );
277 
278  xsysview_.runq = sched2_.running-1;
279  xsysview_.blockq = sched2_.blocked;
280 
281  xsysview_.mem_total = vmstat2_.mem_total;
282  xsysview_.mem_unused = vmstat2_.nr_free_pages * xsysview_.pagesize_;
283  xsysview_.mem_commitas = vmstat2_.committed_as;
284  xsysview_.mem_anon = vmstat2_.nr_anon_pages * xsysview_.pagesize_;
285  xsysview_.mem_file = vmstat2_.nr_file_pages * xsysview_.pagesize_;
286  xsysview_.mem_shmem = vmstat2_.nr_shmem * xsysview_.pagesize_;
287  xsysview_.mem_slab = (vmstat2_.nr_slab_reclaimable + vmstat2_.nr_slab_unreclaimable ) * xsysview_.pagesize_;
288  xsysview_.mem_pagetbls = vmstat2_.nr_page_table_pages * xsysview_.pagesize_;
289  xsysview_.mem_dirty = vmstat2_.nr_dirty * xsysview_.pagesize_;
290  xsysview_.mem_pageins = (double)(vmstat2_.pgpgin - vmstat1_.pgpgin ) / dt;
291  xsysview_.mem_pageouts = (double)(vmstat2_.pgpgout - vmstat1_.pgpgout ) / dt;
292  xsysview_.mem_swapins = (double)(vmstat2_.pswpin - vmstat1_.pswpin ) / dt;
293  xsysview_.mem_swapouts = (double)(vmstat2_.pswpout - vmstat1_.pswpout ) / dt;
294  xsysview_.mem_hptotal = vmstat2_.hugepages_total * vmstat2_.hugepagesize;
295  xsysview_.mem_hprsvd = vmstat2_.hugepages_reserved * vmstat2_.hugepagesize;
296  xsysview_.mem_hpfree = vmstat2_.hugepages_free * vmstat2_.hugepagesize;
297  xsysview_.mem_thpanon = vmstat2_.nr_anon_transparent_hugepages * vmstat2_.hugepagesize;
298  xsysview_.mem_mlock = vmstat2_.nr_mlock * xsysview_.pagesize_;
299  xsysview_.mem_mapped = vmstat2_.nr_mapped * xsysview_.pagesize_;
300  xsysview_.mem_swpused = 0;
301  xsysview_.mem_swpsize = 0;
302  for ( std::list<vmem::SwapInfo>::const_iterator i = swaps_.begin(); i != swaps_.end(); i++ ) {
303  xsysview_.mem_swpused += (*i).used;
304  xsysview_.mem_swpsize += (*i).size;
305  }
306 
307  xsysview_.mem_minflts = (double)(vmstat2_.pgfault - vmstat1_.pgfault ) / dt;
308  xsysview_.mem_majflts = (double)(vmstat2_.pgmajfault - vmstat1_.pgmajfault ) / dt;
309  xsysview_.mem_allocs = (double)(vmstat2_.pgalloc_dma - vmstat1_.pgalloc_dma +
310  vmstat2_.pgalloc_dma32 - vmstat1_.pgalloc_dma32 +
311  vmstat2_.pgalloc_normal - vmstat1_.pgalloc_normal +
312  vmstat2_.pgalloc_movable - vmstat1_.pgalloc_movable ) / dt;
313  xsysview_.mem_frees = (double)(vmstat2_.pgfree - vmstat1_.pgfree ) / dt;
