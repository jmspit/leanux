110  std::sort( vdigits.begin(), vdigits.end() );
111 
112  double total = getCPUUsageTotal( stat );
113 
114  double line_frac = maxvalue/ (double)maxlines;
115 
116  int lines = std::min( maxlines, (int)round( total / line_frac ) );
117  for ( int i = 0; i < lines; i++ ) {
118  vdigits[0].remain_ -= line_frac;
119  vdigits[0].height_++;
120  s += vdigits[0].symbol_;
121  sort( vdigits.begin(), vdigits.end() );
122  };
123  sort( s.begin(), s.end() );
