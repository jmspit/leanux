143  void getSwapInfo( std::list<SwapInfo> &swaps ) {
144  swaps.clear();
145  std::ifstream fs( "/proc/swaps" );
146  std::string s;
147  getline( fs, s );
148  SwapInfo inf;
149  while ( fs.good() ) {
150  fs >> inf.devicefile;
151  fs >> inf.devicetype;
152  fs >> inf.size;
153  fs >> inf.used;
154  fs >> inf.priority;
155  inf.size *= 1024;
156  inf.used *= 1024;
157  if ( fs.good() ) swaps.push_back( inf );
158  }
159  }

177  double getDirtyPressure( ssize_t ram, ssize_t dirtied ) {
178  ssize_t dirty_limit = getDirtyBytes();
179  if ( dirty_limit == 0 ) dirty_limit = (double)ram / 100.0 * (double)getDirtyRatio();
