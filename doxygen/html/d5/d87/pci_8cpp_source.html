99  PCIAddress decodeEBDF( const std::string &s ) {
100  PCIAddress a = NullPCIAddress;
101  std::stringstream ss;
102  ss.str(s);
103  ss >> std::hex;
104  ss >> a.domain;
105  if ( !ss.good() ) return NullPCIAddress;
106  ss.get();
107  ss >> a.bus;
108  if ( !ss.good() ) return NullPCIAddress;
109  ss.get();
110  ss >> a.device;
111  if ( !ss.good() ) return NullPCIAddress;
112  ss.get();
113  ss >> a.function;
114  if ( !ss.good() ) return NullPCIAddress;
115  return a;
116  }

510  bool getPCIHardwareInfo( const PCIHardwareId &id, PCIHardwareInfo &inf ) {
511  inf.vendor = "";
512  inf.device = "";
513  inf.subdevice = "";
514  inf.pciclass = getPCIHardwareClassString( id.pciclass );
515  inf.pcisubclass = getPCISubClassString( id.pciclass );
516  std::stringstream ss;
517  ss << std::hex;
518  ss << std::setfill('0') << std::setw(4) << id.vendor;
519  std::string hex_vendor = ss.str();
520  ss.str("");
521  ss << std::setfill('0') << std::setw(4) << id.device;
522  std::string hex_device = ss.str();
523  ss.str("");
524  ss << std::setfill('0') << std::setw(4) << id.subvendor << ' ' << id.subdevice;
525  std::string hex_subdevice = ss.str();
526 
527  std::ifstream f( PCIDeviceDatabase.c_str() );
528  if ( !f.good() ) throw Oops( __FILE__, __LINE__, "error opening pci.ids" );
529 
536  int state = 0;
537 
538  bool giveup = false;
539  while ( f.good() && !f.eof() && !giveup && state != 3 ) {
540  std::string s = "";
541  getline( f, s );
542 
543  if ( s[0] != '#' ) {
544 
545  switch ( state ) {
546  case 0:
547  if ( strncmp( s.c_str(), hex_vendor.c_str(), 4 ) == 0 ) {
548  inf.vendor = s.substr( 6 );
549  state = 1;
550  }
551  break;
552  case 1:
553  if ( s[0] != '\t' ) {
554  //we hit another vendor line, so device not found
555  giveup = true;
556  } else {
557  if ( strncmp( s.c_str()+1, hex_device.c_str(), 4 ) == 0 ) {
558  inf.device = s.substr(7);
559  state = 2;
560  }
561  }
562  break;
563  case 2:
564  if ( s[0] != '\t' || s[1] != '\t' ) {
565  //possible subvendors exhausted, giveup
566  giveup = true;
567  } else {
568  if ( strncmp( s.c_str()+2, hex_subdevice.c_str(), 9 ) == 0 ) {
569  inf.subdevice = s.substr(13);
570  state = 3;
571  }
572  }
573  break;
574 
575  }
576  }
577  }
578  return ( state == 2 || state == 3 );
579  }

584  void enumPCIDevicePaths( std::list<PCIDevicePath> &paths ) {
585  paths.clear();
586  DIR *d;
587  struct dirent *dir;
588  d = opendir( (sysdevice::sysbus_root + "/pci/devices").c_str() );
589  if ( d ) {
590  while ( (dir = readdir(d)) != NULL ) {
591  if ( dir->d_name[0] != '.' ) {
592  std::string path = sysdevice::sysbus_root + "/pci/devices/" + (std::string)dir->d_name;
593  std::string resolved_path = util::realPath( path );
594  if ( resolved_path.length() > 13 ) paths.push_back( resolved_path.substr(13) );
595  }
596  }
597  closedir( d );
598  }
599  }
