282  bool isWholeDisk() const {
283  if ( isSCSIDisk( MAJOR(dev_) ) ) return (MINOR(dev_) % 16) == 0;
284  else if ( isIDEDisk( MAJOR(dev_) ) ) return (MINOR(dev_) % 64 ) == 0;
285  else if ( isVirtIODisk( *this ) && MINOR(dev_) % 16 == 0 ) return true;
286  else if ( isNVMeDisk( *this ) && !isPartition() ) return true;
287  else if ( isMMCDisk( *this ) && !isPartition() ) return true;
288  else return false;
289  }
290 
294  bool isPartition() const {
295  if ( isSCSIDisk( MAJOR(dev_) ) ) return (MINOR(dev_) % 16) != 0;
296  else if ( isIDEDisk( MAJOR(dev_) ) ) return (MINOR(dev_) % 64 ) != 0;
297  else if ( isVirtIODisk( MajorMinor( MAJOR(dev_), 0 ) ) ) return (MINOR(dev_) % 16 ) != 0;
298  else if ( isNVMeDisk( MajorMinor( MAJOR(dev_), 0 ) ) ) {
299  unsigned int host = 0;
300  unsigned int disk = 0;
301  unsigned int partition = 0;
302  std::string devname = getNameByMajorMinor( *this );
303  int r = sscanf( devname.c_str(), "nvme%un%up%u", &host, &disk, &partition );
304  return r == 3;
305  }
306  else if ( isMMCDisk( MajorMinor( MAJOR(dev_), MINOR(dev_) ) ) ) {
307  unsigned int disk = 0;
308  unsigned int partition = 0;
309  std::string devname = getNameByMajorMinor( * this );
310  int r = sscanf( devname.c_str(), "mmcblk%up%u", &disk, &partition );
311  return r == 2;
312  }
313  else if ( isBCacheDisk( MajorMinor( MAJOR(dev_), MINOR(dev_) ) ) ) {
314  // it seems creating partitions on bcache devices does not really work
315  // but if it would/will it would look like this
316  unsigned int disk = 0;
317  unsigned int partition = 0;
318  std::string devname = getNameByMajorMinor( * this );
319  int r = sscanf( devname.c_str(), "bcache%up%u", &disk, &partition );
320  return r == 2;
321  }
322  else return false;
323  }
