56  time_t LardHistory::setZoom( time_t zoom ) {
57  if ( snap_range_ == 0 && zoom < zoom_ ) return zoom_;
58  if ( snap_range_ == snap_max_ - snap_min_ && zoom > zoom_ ) return zoom_;
59  zoom_ = zoom;
60  updateMinMax();
61  if ( snap_start_ + snap_range_ <= snap_max_ ) {
62  snap_end_ = snap_start_ + snap_range_;
63  } else {
64  snap_end_ = snap_max_;
65  snap_start_ = snap_end_ - snap_range_;
66  }
67  updateRangeTime();
68  cpumap_.clear(); // wipe the cache
69  return zoom_;
70  }

123  updateMinMax();
124  long diff = 3600.0/(double)(snap_end_time_ - snap_start_time_)*(snap_range_+1);
125  if ( snap_end_ + diff + 1 <= snap_max_ ) {
126  snap_start_ += (diff);
127  snap_end_ += (diff);
128  updateRangeTime();
129  cpumap_.clear();

131  snap_end_ = snap_max_;
132  snap_start_ = snap_end_ - snap_range_;
133  updateRangeTime();
134  cpumap_.clear();
135  }

234  void LardHistory::updateRangeTime() {
235  persist::Query qrange( *db_ );
236  qrange.prepare( "select istart from snapshot where id=:min" );
237  qrange.bind( 1, snap_start_ );
238  if ( qrange.step() ) {
239  snap_start_time_ = qrange.getLong(0);
240  }
241  qrange.prepare( "select istop from snapshot where id=:max" );
242  qrange.bind( 1, snap_end_ );
243  if ( qrange.step() ) {
244  snap_end_time_ = qrange.getLong(0);

249  persist::Query qminmax( *db_ );
250  qminmax.prepare( "select min(id), max(id),min(istart),max(istop), avg(istop-istart) from snapshot" );
251  if ( qminmax.step() ) {
252  snap_min_ = qminmax.getLong( 0 );
253  snap_max_ = qminmax.getLong( 1 );
254  snap_min_start_time_ = qminmax.getLong( 2 );
255  snap_max_stop_time_ = qminmax.getLong( 3 );
256  snap_avg_time_ = qminmax.getDouble( 4 );
257  snap_range_ = zoom_ / qminmax.getDouble( 4 ) - 1;
258  if ( snap_range_ < 0 ) snap_range_ = 0;
259  if ( snap_range_ > snap_max_ - snap_min_ ) snap_range_ = snap_max_ - snap_min_;
260  }

334 
335  stat.state='S';
336  stat.wchan="";
337  qstate.reset();
338  qstate.bind( 1, stat.pid );
339  qstate.bind( 2, snap_min_ );
340  qstate.bind( 3, snap_max_ );
341  if ( qstate.step() ) {
342  if ( !qstate.isNull(0) ) {
343  stat.state = qstate.getText(0)[0];
344  stat.wchan = qstate.getText(1);
345  }
346  }
347 
348  procview.delta.push_back(stat);
349  }
350  procview.sample_count = 2;
351  }
