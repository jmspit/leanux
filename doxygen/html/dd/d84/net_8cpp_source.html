123  void getMACOUI( const std::string &mac, OUI &oui ) {
124  oui.vendor = "";
125  oui.countrycode = "";
126  oui.address.clear();
127  std::string search = mac.substr(0,2) + mac.substr(3,2) + mac.substr(6,2);
128  std::transform(search.begin(), search.end(), search.begin(), toupper);
129 
130  std::ifstream f( MACOUIDatabase.c_str() );
131  if ( !f.good() ) throw Oops( __FILE__, __LINE__, "error opening oui.txt" );
132 
133  while ( f.good() && !f.eof() ) {
134  std::string s = "";
135  getline( f, s );
136  if ( strncmp( search.c_str(), s.c_str(), 6 ) == 0 ) {
137  size_t lasttab = s.find_last_of( '\t', std::string::npos );
138  if ( lasttab == std::string::npos || lasttab >= s.length() - 1 )
139  throw Oops( __FILE__, __LINE__, "parse error in oui.txt" );
140  else {
141  // the remainder of the first string is the vendor
142  oui.vendor = s.substr( lasttab + 1 );
143  }
144  std::vector<std::string> tmp;
145  while ( f.good() && tmp.size() < 10 ) {
146  getline( f, s );
147  if ( s == "" ) break;
148  lasttab = s.find_last_of( '\t', std::string::npos );
149  if ( lasttab != std::string::npos ) tmp.push_back( s.substr(lasttab+1) ); else break;
150  }
151  if ( tmp.size() < 3 ) throw Oops( __FILE__, __LINE__, "parse error in oui.txt" );
152  for ( size_t j = 0; j < tmp.size()-1; j++ ) {
153  oui.address.push_back( tmp[j] );
154  }
155  oui.countrycode = tmp[tmp.size()-1];
156  break;
157  }
158  }
159  }

384  std::string resolveIP( const std::string &addr ) {
385  if ( whatAF_INET( addr ) == AF_INET ) {
386  struct sockaddr_in sa;
387  sa.sin_family = AF_INET;
388  int r = inet_pton( AF_INET, addr.c_str(), &sa.sin_addr );
389  if ( r != 1 ) throw Oops( __FILE__, __LINE__, "not a valid ip4 address" );
390  char hbuf[NI_MAXHOST];
391  memset( hbuf, '\0', sizeof(hbuf) );
392  r= getnameinfo( (struct sockaddr*)&sa, sizeof(sa), hbuf, sizeof(hbuf), NULL, 0, NI_NAMEREQD );
393  if ( r == 8 ) throw Oops( __FILE__, __LINE__, gai_strerror(r) );
394  else if ( r == EAI_NONAME ) return addr;
395  return hbuf;
396  } else if ( whatAF_INET( addr ) == AF_INET6 ) {
397  struct sockaddr_in6 sa;
398  sa.sin6_family = AF_INET6;
399  int r = inet_pton( AF_INET6, addr.c_str(), &sa.sin6_addr );
400  if ( r != 1 ) throw Oops( __FILE__, __LINE__, "not a valid ipv6 address" );
401  char hbuf[NI_MAXHOST];
402  memset( hbuf, '\0', sizeof(hbuf) );
403  r= getnameinfo( (struct sockaddr*)&sa, sizeof(sa), hbuf, sizeof(hbuf), NULL, 0, NI_NAMEREQD );
404  if ( r ) {
405  if ( r == EAI_NONAME ) return addr; //else throw Oops( __FILE__, __LINE__, gai_strerror(r) );
406  }
407  return hbuf;
408  }
409  return "";
410  }

558  void getNetStatDelta( const NetStatDeviceMap& snap1, const NetStatDeviceMap& snap2, NetStatDeviceVector& delta ) {
559  delta.clear();
560  for ( NetStatDeviceMap::const_iterator s2 = snap2.begin(); s2 != snap2.end(); ++s2 ) {

562  NetStat d;
