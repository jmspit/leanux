58  bool getProcPidStat( pid_t pid, ProcPidStat &stat ) {
59  std::stringstream path;
60  stat.pid = pid;
61  stat.comm = "";
62  path << "/proc/" << pid << "/task/" << pid << "/stat";
63  std::ifstream ifs( path.str().c_str() );
64  std::string s;
65  getline( ifs, s );
66  if ( ifs.good() ) {
67  size_t p = 0, q = 0;
68  for ( p = 0; p < s.length(); p++ ) {
69  if ( !(isdigit( s[p] ) || s[p] == ' ' ) ) {
70  if ( s[p] != '(' ) throw Oops( __FILE__, __LINE__, "parse failure on " + path.str() ); else {
71  //here begins the 'cmd' field
72  for ( q = p+1; q < s.length() && (s[q] != ')' || (q<s.length()-1 && s[q+1] == ')')); q++ ) {
73  stat.comm += s[q];
74  }
75  p = q+2;
76  break;
77  }
78  }
79  }
80  unsigned long utime;
81  unsigned long stime;
82  unsigned long cutime;
83  unsigned long cstime;
84  unsigned long long delayacct_blkio_ticks;
85  if ( 21 <= sscanf( s.substr(p).c_str(), "%c %d %d %d %d %d %*u %lu %lu %lu %lu %lu %lu %ld %ld %ld %ld %ld %*d %llu %lu %ld %lu"
86  " %*u %*u %*u %*u %*u %*u %*u %*u %*u %*u %*u %*u %*d" // ignore 26-startcode to 38-exit_signal
87  "%d %*u %*u %llu",
88  &stat.state,
89  &stat.ppid,
90  &stat.pgrp,
91  &stat.session,
92  &stat.tty_nr,
93  &stat.tpgid,
94  &stat.minflt,
95  &stat.cminflt,
96  &stat.majflt,
97  &stat.cmajflt,
98  &utime,
99  &stime,
100  &cutime,
101  &cstime,
102  &stat.priority,
103  &stat.nice,
104  &stat.num_threads,
105  &stat.starttime,
106  &stat.vsize,
107  &stat.rss,
108  &stat.rsslim,
109  &stat.processor,
110  &delayacct_blkio_ticks ) ) {
111  stat.wchan = process::getWChan( pid );
112  stat.utime = utime / (double)leanux::system::getUserHz();
113  stat.stime = stime / (double)leanux::system::getUserHz();
114  stat.cutime = cutime / (double)leanux::system::getUserHz();
115  stat.cstime = cstime / (double)leanux::system::getUserHz();
116  stat.delayacct_blkio_ticks = (double)delayacct_blkio_ticks / (double)leanux::system::getUserHz();
117  } else throw Oops( __FILE__, __LINE__, "parse failure on " + path.str() );
118  } else return false;
119  return true;
120  };

136  bool getProcPidIO( pid_t pid, ProcPidIO &io ) {\
137  std::stringstream ss;
138  ss << "/proc/" << pid << "/io";
139  std::ifstream ifs( ss.str().c_str() );
140  if ( ifs.good() ) {
141  while ( ifs.good() && !ifs.eof() ) {
142  std::string s;
143  getline( ifs, s );
144  unsigned long t;
145  if ( sscanf( s.c_str(), "rchar: %lu", &t ) == 1 ) io.rchar = t;
146  else if ( sscanf( s.c_str(), "wchar: %lu", &t ) == 1 ) io.wchar = t;
147  else if ( sscanf( s.c_str(), "syscr: %lu", &t ) == 1 ) io.syscr = t;
148  else if ( sscanf( s.c_str(), "syscw: %lu", &t ) == 1 ) io.syscw = t;
149  else if ( sscanf( s.c_str(), "read_bytes: %lu", &t ) == 1 ) io.read_bytes = t;
150  else if ( sscanf( s.c_str(), "write_bytes: %lu", &t ) == 1 ) io.write_bytes = t;
151  else if ( sscanf( s.c_str(), "cancelled_write_bytes: %lu", &t ) == 1 ) io.cancelled_write_bytes = t;
152  }
153  } else return false;
154  return true;
155  }

173  void getAllProcPidStat( ProcPidStatMap &stats ) {
174  stats.clear();
175  std::set<pid_t> threadset;
176  std::string path = "/proc";
177  DIR *pidd;
178  struct dirent *piddir;
179  pidd = opendir( path.c_str() );
180  if ( pidd ) {
181  while ( ( piddir = readdir( pidd )) != NULL ) {
182  if ( isdigit( piddir->d_name[0] ) ) {
183  pid_t pid = atoi( piddir->d_name );
184  if ( pid ) {
185  std::string tpath = "/proc/" + (std::string)piddir->d_name + "/task";
186  DIR *tidd;
187  struct dirent *tiddir;
188  tidd = opendir( tpath.c_str() );
189  if ( tidd ) {
190  while ( ( tiddir = readdir( tidd )) != NULL ) {
191  if ( isdigit( tiddir->d_name[0] ) ) {
192  pid_t tid = atoi( tiddir->d_name );
193  threadset.insert(tid);
194  }
195  }
196  }
197  closedir( tidd );
198  threadset.insert(pid);
199  }
200  }
201  }
202  } else throw Oops( __FILE__, __LINE__, errno );
203  closedir( pidd );
204  for ( std::set<pid_t>::const_iterator t = threadset.begin(); t != threadset.end(); ++t ) {
205  ProcPidStat stat;
206  if ( getProcPidStat( *t, stat ) ) {
207  stats[*t] = stat;
208  }
209  }
210  };

244  void deltaProcPidStats( const ProcPidStatMap &snap1, const ProcPidStatMap &snap2, ProcPidStatDeltaVector &delta ) {
245  delta.clear();
246  for ( ProcPidStatMap::const_iterator s2 = snap2.begin(); s2 != snap2.end(); ++s2 ) {
247  ProcPidStatDelta dt;
248  dt.pid = s2->first;

251  if ( s1 != snap1.end() ) {
252  dt.utime = s2->second.utime - s1->second.utime;
253  dt.stime = s2->second.stime - s1->second.stime;
254  dt.minflt = s2->second.minflt - s1->second.minflt;
255  dt.majflt = s2->second.majflt - s1->second.majflt;
256  dt.delayacct_blkio_ticks = s2->second.delayacct_blkio_ticks - s1->second.delayacct_blkio_ticks;
257  dt.state = s2->second.state;
258  dt.rss = s2->second.rss;
259  dt.vsize = s2->second.vsize;
260  dt.comm = s2->second.comm;
261  dt.wchan = s2->second.wchan;
262  delta.push_back( dt );
263  } else {
264  // with only end-interval statistics...we return simply the stats of snap2
265  dt.utime = s2->second.utime;
266  dt.stime = s2->second.stime;
267  dt.minflt = s2->second.minflt;

293  int StatsSorter::operator()( ProcPidStatDelta d1, ProcPidStatDelta d2 ) {
294  switch ( sortby_ ) {
295  case utime:
296  return
297  (d1.utime > d2.utime) ||
298  (d1.utime == d2.utime && d1.stime > d2.stime ) ||
299  (d1.utime == d2.utime && d1.stime == d2.stime && d1.pid > d2.pid );
300  case stime:
301  return d1.stime > d2.stime;
302  case cputime:
303  return d1.utime+d1.stime > d2.utime + d2.stime;
304  case minflt:
305  return d1.minflt > d2.minflt;
306  case majflt:
307  return d1.majflt > d2.majflt;
308  case rss:
309  return d1.rss > d2.rss;
310  case vsize:
311  return d1.vsize > d2.vsize;
312  case top:
313  //return ( d1.utime+d1.stime+d1.delayacct_blkio_ticks > d2.utime+d2.stime+d2.delayacct_blkio_ticks );
314  default:
315  return ( d1.utime+d1.stime+d1.delayacct_blkio_ticks > d2.utime+d2.stime+d2.delayacct_blkio_ticks ) ||
316  ( d1.utime+d1.stime+d1.delayacct_blkio_ticks == d2.utime+d2.stime+d2.delayacct_blkio_ticks && d1.majflt > d2.majflt ) ||
317  ( d1.utime+d1.stime+d1.delayacct_blkio_ticks == d2.utime+d2.stime+d2.delayacct_blkio_ticks && d1.majflt == d2.majflt && d1.minflt > d2.minflt ) ||
318  ( d1.utime+d1.stime+d1.delayacct_blkio_ticks == d2.utime+d2.stime+d2.delayacct_blkio_ticks && d1.majflt == d2.majflt && d1.minflt == d2.minflt && d1.rss > d2.rss);
