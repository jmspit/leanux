289  bool VirtioBlockDevice::accept( SysDevicePath &path ) {
290  if ( path.find("virtio") == std::string::npos ) return false;
291  std::list<std::string> tokens;
292  tokenize( path, tokens );
293  const unsigned int st_blkdev_name = 0; // <vda>
294  const unsigned int st_const_block = 1; // block
295  const unsigned int st_virtio_version = 2; // virtio<version>
296  const unsigned int st_accept = 3;
297  const unsigned int st_fail = 4;
298  unsigned int state = st_blkdev_name;
299  util::RegExp reg;
300  size_t eat_chars = 0;
301  for ( std::list<std::string>::const_iterator t = tokens.begin(); t != tokens.end() && state != st_accept && state != st_fail; t++ ) {
302  switch ( state ) {
303  case st_blkdev_name:
304  reg.set( "(vd).*" );
305  if ( reg.match( *t ) ) {
306  state = st_const_block;
307  eat_chars += (*t).length() + 1;
308  } else state = st_fail;
309  break;
310  case st_const_block:
311  if ( *t == "block" ) {
312  state = st_virtio_version;
313  eat_chars += (*t).length() + 1;
314  } else state = st_fail;
315  break;
316  case st_virtio_version:
317  reg.set( "(virtio)([[:digit:]]+)" );
318  if ( reg.match( *t ) ) {
319  state = st_accept;
320  eat_chars += (*t).length() + 1;
321  } else state = st_fail;
322  break;
323  }
324  }
325  if ( state == st_accept ) {
326  path_ = path;
327  leaf_ = path.substr( path.length() - eat_chars + 1 );
328  path = path.substr( 0, path.length() - eat_chars );
329  return true;
330  } else return false;
331  }
