245  void Screen::screenResize() {
246 
247  const int idx_header = 0;
248  const int idx_cpu = 1;
249  const int idx_disk = 2;
250  const int idx_net = 3;
251  const int idx_proc = 4;
252  const int idx_foot = 5;
253  int lines[6];
254  lines[idx_header] = Header::getMinHeight();
255  lines[idx_cpu] = SysView::getMinHeight();
256  lines[idx_disk] = IOView::getMinHeight();
257  lines[idx_net] = NetView::getMinHeight();
258  lines[idx_proc] = ProcessView::getMinHeight();
259  lines[idx_foot] = Footer::getMinHeight();
260  int surplus = winsz_.ws_row;
261  for ( int i = 0; i < 6; i++ ) surplus -= lines[i];
262  int cur_idx = idx_header;
263  while ( surplus > 0 ) {
264  switch ( cur_idx ) {
265  case idx_header:
266  if ( lines[idx_header] < Header::getOptimalHeight() ) {
267  lines[idx_header]++;
268  surplus--;
269  }
270  break;
271  case idx_cpu:
272  if ( lines[idx_cpu] < SysView::getOptimalHeight() ) {
273  lines[idx_cpu]++;
274  surplus--;
275  }
276  break;
277  case idx_disk:
278  if ( lines[idx_disk] < IOView::getOptimalHeight() ) {
279  lines[idx_disk]++;
280  surplus--;
281  }
282  break;
283  case idx_net:
284  if ( lines[idx_net] < NetView::getOptimalHeight() ) {
285  lines[idx_net]++;
286  surplus--;
287  }
288  break;
289  case idx_proc:
290  if ( lines[idx_proc] < ProcessView::getOptimalHeight() ) {
291  lines[idx_proc]++;
292  surplus--;
293  }
294  break;
295  case idx_foot:
296  if ( lines[idx_foot] < Footer::getOptimalHeight() ) {
297  lines[idx_foot]++;
298  surplus--;
299  }
300  break;
301  }
302  cur_idx++;
303  if ( cur_idx >= 6 ) cur_idx = 0;
304  }
305 
306  #ifdef TRACEON
307  std::stringstream s;
308  s.str("");
309  s << "lines=" << lines[idx_header] << " "
310  << lines[idx_cpu] << " "
311  << lines[idx_disk] << " "
312  << lines[idx_net] << " "
313  << lines[idx_proc] << " "
314  << lines[idx_foot] << " total "
315  << lines[idx_header]+lines[idx_cpu]+lines[idx_disk]+lines[idx_cpu]+lines[idx_net] + lines[idx_proc] + lines[idx_foot]<< std::endl;
316  TRACE( s );
317  #endif
318 
319  if ( !vheader_ )
320  vheader_ = new Header( winsz_.ws_col, Header::getMinHeight(), 0, 0, this );
321  else
322  vheader_->resize( winsz_.ws_col, Header::getMinHeight(), 0, 0 );
323  if ( !vsys_ )
324  vsys_ = new SysView( winsz_.ws_col, lines[idx_cpu], 0, Header::getMinHeight(), this );
325  else
326  vsys_->resize( winsz_.ws_col, lines[idx_cpu], 0, lines[idx_header] );
327  if ( !vio_ )
328  vio_ = new IOView( winsz_.ws_col, lines[idx_disk], 0, lines[idx_header] + lines[idx_cpu], this );
329  else
330  vio_->resize( winsz_.ws_col, lines[idx_disk], 0, lines[idx_header] + lines[idx_cpu] );
331  if ( !vnetwork_ )
332  vnetwork_ = new NetView( winsz_.ws_col, lines[idx_net], 0, lines[idx_header] + lines[idx_cpu] + lines[idx_disk], this );
333  else
334  vnetwork_->resize( winsz_.ws_col, lines[idx_net], 0, lines[idx_header] + lines[idx_cpu] + lines[idx_disk] );
335  if ( !vprocess_ )
336  vprocess_ = new ProcessView( winsz_.ws_col, lines[idx_proc], 0, lines[idx_header] + lines[idx_cpu] + lines[idx_disk] + lines[idx_net], this );
337  else
338  vprocess_->resize( winsz_.ws_col, lines[idx_proc], 0, lines[idx_header] + lines[idx_cpu] + lines[idx_disk] + lines[idx_net] );
339  if ( !vfooter_ )
340  vfooter_ = new Footer( winsz_.ws_col, Footer::getMinHeight(), 0, lines[idx_header] + lines[idx_cpu] + lines[idx_disk] + lines[idx_net] + lines[idx_proc], this );
341  else {
342  vfooter_->resize( winsz_.ws_col, Footer::getMinHeight(), 0, lines[idx_header] + lines[idx_cpu] + lines[idx_disk] + lines[idx_net] + lines[idx_proc] );
343  }
344  }

353  void Screen::runHistory( persist::Database *db ) {
354  reportMessage( HLP_QUIT, 0, "press q or ^C to quit" );
355  reportMessage( HLP_BROWSE_ARROW, 0, "move back/forward leftarrow/rightarrow or ,/. " );
356  reportMessage( HLP_BROWSE_HOUR, 0, "move back/forward one hour h/H" );
357  reportMessage( HLP_BROWSE_DAY, 0, "move back/forward one day d/D" );
358  reportMessage( HLP_BROWSE_WEEK, 0, "move back/forward one week w/W" );
359  reportMessage( HLP_BROWSE_HOMEND, 0, "move to front/end of history HOME/END" );
360  sleep_duration_ms_ = 20;
361 
362  time_t cur_zoom = 300;
363 
364  LardHistory history( db, cur_zoom );
365  XSysView sysview;
366  XIOView ioview;
367  XNetView netview;
368  XProcView procview;
369  stopped_ = false;
370  bool update_required = true;
371  bool footer_refresh = true;
372  struct timeval samplet1, samplet2, scrollt1, scrollt2, last_size_check;
373  gettimeofday( &samplet1, 0 );
374  gettimeofday( &samplet2, 0 );
375  scrollt1 = samplet1;
376  scrollt2 = samplet2;
377 
378  gettimeofday( &last_size_check, 0 );
379 
380  while ( !stopped_ ) {
381 
382  update_required = true;
383  int key = getch();
384  if ( key == 'q' ) {
385  stopped_ = true;
386  update_required = false;
387  } else if ( key == KEY_RIGHT ) {
388  history.rangeUp();
389  } else if ( key == KEY_LEFT ) {
390  history.rangeDown();
391  } else if ( key == KEY_HOME ) {
392  history.rangeStart();
393  } else if ( key == KEY_END ) {
394  history.rangeEnd();
395  } else if (key == '-' ) {
396  cur_zoom = history.zoomOut();
397  } else if ( key == '+' ) {
398  cur_zoom = history.zoomIn();
399  } else if ( key == 'h' ) {
400  history.hourDown();
401  } else if ( key == 'H' ) {
402  history.hourUp();
403  } else if ( key == 'd' ) {
404  history.dayDown();
405  } else if ( key == 'D' ) {
406  history.dayUp();
407  } else if ( key == 'w' ) {
408  history.weekDown();
409  } else if ( key == 'W' ) {
410  history.weekUp();
411  }
412 
413 
414  if ( update_required ) {
415  history.fetchXSysView( sysview );
416  history.fetchXIOView( ioview );
417  history.fetchXNetView( netview );
418  history.fetchXProcView( procview );
419  }
420 
421  gettimeofday( &samplet2, 0 );
422  if ( util::deltaTime( last_size_check, samplet2 ) >= 0.1 ) {
423  last_size_check = samplet2;
424  if ( sizeChanged() ) {
425  if ( ioctl( 0, TIOCGWINSZ, &winsz_ ) == -1 ) throw Oops( __FILE__, __LINE__, errno );
426  werase(wmain_);
427  wrefresh(wmain_);
428  resize_term(winsz_.ws_row,winsz_.ws_col);
429  screenResize();
430  update_required = true;
431  footer_refresh = true;
432  doupdate();
433  refresh();
434 
435  #ifdef TRACEON
436  std::stringstream s;
437  s << "resized window size col=" << winsz_.ws_col << " row=" << winsz_.ws_row << " " << COLS << "/" << LINES << std::endl;
438  TRACE( s );
439  #endif
440  }
441  }
442 
443  gettimeofday( &scrollt2, 0 );
444  if ( util::deltaTime( scrollt1, scrollt2 ) > footer_scroll_interval_s_ ) {
445  scrollt1 = scrollt2;
446  footer_refresh = true;
447  }
448 
449  if ( update_required || footer_refresh ) {
450  if ( update_required ) {
451  std::stringstream ss;
452  ss << "zoom " << util::TimeStrSec( util::deltaTime( sysview.t1, sysview.t2 ) ) << " "
453  //<< history.getStartSnap() << "/" << history.getEndSnap()
454  //<< " = "
455  << util::localStrISODateTime( history.getStartTime() )
456  << " - " << util::localStrISOTime( history.getEndTime() );
457  ((Header*)vheader_)->xrefresh( "browse " + db->fileName(), ss.str() );
458  ((SysView*)vsys_)->xrefresh( sysview, true );
459  ((IOView*)vio_)->xrefresh( ioview );
460  ((NetView*)vnetwork_)->xrefresh( netview );
461  ((ProcessView*)vprocess_)->xrefresh( procview );
462  update_required = false;
463  }
464  if ( footer_refresh ) {
465  ((Footer*)vfooter_)->xrefresh();
466  footer_refresh = false;
467  }
468  doupdate();
469  }
470 
471  if ( !stopped_ ) util::Sleep( 0, (long int)(sleep_duration_ms_ * 1.0E6) );
472  }

475  void Screen::runRealtime() {
476  reportMessage( HLP_QUIT, 0, "press q or ^C to quit" );
477  reportMessage( HLP_BROWSE_ARROW, 0, "increase/decrease sampling interval -/+" );
478  RealtimeSampler realtimesampler;
479  stopped_ = false;
480  bool update_required = true;
481  bool footer_refresh = true;
482  struct timeval samplet1, samplet2, scrollt1, scrollt2, last_size_check;
483  gettimeofday( &samplet1, 0 );
484  gettimeofday( &samplet2, 0 );
485  scrollt1 = samplet1;
486  scrollt2 = samplet2;
487 
488  gettimeofday( &last_size_check, 0 );
489 
490  while ( !stopped_ ) {
491 
492  // any recognized user input?
493  int key = getch();
494  if ( key == 'q' ) {
495  stopped_ = true;
496  } else if ( key == '+' ) {
497  sample_interval_s_ += 1;
498  realtimesampler.resetCPUTrail();
499  update_required = true;
500  } else if ( key == '-' ) {
501  if ( sample_interval_s_ > 1 ) {
502  sample_interval_s_ -= 1;
503  realtimesampler.resetCPUTrail();
504  update_required = true;
505  }
506  }
507 
508  gettimeofday( &samplet2, 0 );
509  if ( util::deltaTime( last_size_check, samplet2 ) >= 0.1 ) {
510  last_size_check = samplet2;
511  if ( sizeChanged() ) {
512  if ( ioctl( 0, TIOCGWINSZ, &winsz_ ) == -1 ) throw Oops( __FILE__, __LINE__, errno );
513  werase(wmain_);
514  wrefresh(wmain_);
515  resize_term(winsz_.ws_row,winsz_.ws_col);
516  screenResize();
517  update_required = true;
518  footer_refresh = true;
519  doupdate();
520  refresh();
521 
522  #ifdef TRACEON
523  std::stringstream s;
524  s << "resized window size col=" << winsz_.ws_col << " row=" << winsz_.ws_row << " " << COLS << "/" << LINES << std::endl;
525  TRACE( s );
526  #endif
527  }
528  }
529  if ( util::deltaTime( samplet1, samplet2 ) > sample_interval_s_ ) {
530  samplet1 = samplet2;
531  realtimesampler.sample( vsys_->getHeight() );
532  update_required = true;
533  }
534 
535  gettimeofday( &scrollt2, 0 );
536  if ( util::deltaTime( scrollt1, scrollt2 ) > footer_scroll_interval_s_ ) {
537  scrollt1 = scrollt2;
538  footer_refresh = true;
539  }
540 
541  if ( update_required || footer_refresh ) {
542  if ( update_required ) {
543  std::stringstream ss;
544  ss << "each " << sample_interval_s_ << "s " << util::localStrISODateTime();
545  ((Header*)vheader_)->xrefresh( "realtime", ss.str() );
546  ((SysView*)vsys_)->xrefresh( realtimesampler.getXSysView(), false );
547  ((IOView*)vio_)->xrefresh( realtimesampler.getXIOView() );
548  ((NetView*)vnetwork_)->xrefresh( realtimesampler.getXNetView() );
549  ((ProcessView*)vprocess_)->xrefresh( realtimesampler.getXProcView() );
550  update_required = false;
551  }
552  if ( footer_refresh ) {
553  ((Footer*)vfooter_)->xrefresh();
554  footer_refresh = false;
555  }
556  doupdate();
557  }
558 
559  if ( !stopped_ ) util::Sleep( 0, (long int)(sleep_duration_ms_ * 1.0E6) );
560  }
