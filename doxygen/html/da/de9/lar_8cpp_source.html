134  long ensureSinglePid( persist::Database &db ) {
135  try {
136  persist::Query qry(db);
137  qry.prepare( "SELECT count(name) FROM sqlite_master WHERE type='table' AND name='status'" );
138  if ( qry.step() ) {
139  if ( qry.getInt(0) == 0 ) {
140  createTableStatus( db );
141  }
142  }
143  qry.reset();
144 
145  qry.prepare( "SELECT value FROM status WHERE name=:name" );
146  qry.bind(1,pid_name);
147  if ( qry.step() ) {
148  long db_pid = qry.getInt(0);
149  process::ProcPidStat stat;
150  if ( process::getProcPidStat( db_pid, stat ) ) {
151  if ( stat.comm == "lard" ) {
152  return db_pid;
153  }
154  }
155  persist::DML dml(db);
156  dml.prepare( "UPDATE status SET value=:value WHERE name=:name" );
157  dml.bind(1,getpid());
158  dml.bind(2,pid_name);
159  dml.execute();
160  dml.close();
161  } else {
162  persist::DML dml(db);
163  dml.prepare( "INSERT INTO status (name,value) VALUES (:name,:value)" );
164  dml.bind( 1, pid_name );
165  dml.bind( 2, getpid() );
166  dml.execute();
167  dml.close();
168  }
169  }
170  catch ( const Oops &oops ) {
171  sysLog( LOG_ERR, util::ConfigFile::getConfig()->getIntValue("LOG_LEVEL"), oops.getMessage() );
172  }
173  return 0;
174  }

176  void mainLoop() {
177 
178  persist::Database::softHeapLimit(util::ConfigFile::getConfig()->getIntValue("SQLITE_SOFT_HEAPLIMIT"));
179 
180  initDB();
181  leanux::persist::Database db( options.database, wait_log_handler );
182  long pid = ensureSinglePid( db );
183  if ( pid != 0 ) {
184  std::stringstream ss;
185  ss << "pid " << pid << " is already controlling '" << options.database << "'";
186  sysLog( LOG_ERR, util::ConfigFile::getConfig()->getIntValue("LOG_LEVEL"), ss.str() );
187  return;
188  }
189  {
190  persist::Query query( db );
191  query.prepare( "PRAGMA journal_size_limit=0" );
192  query.step();
193  query.close();
194  }
195  db.enableForeignKeys();
196  createSchema( db );
197  updateSchema( db );
198  std::stringstream ss;
199  ss << "sampling to " << options.database << " version " << db.getUserVersion();
200  sysLog( LOG_STAT, util::ConfigFile::getConfig()->getIntValue("LOG_LEVEL"), ss.str() );
201  double sample_interval = util::ConfigFile::getConfig()->getIntValue("SNAPSHOT_INTERVAL");
202  double maintenance_interval = util::ConfigFile::getConfig()->getIntValue("MAINTENANCE_INTERVAL") * 60;
203  long snapshot_checkpoint = util::ConfigFile::getConfig()->getIntValue("SNAPSHOT_CHECKPOINT");
204 
205  util::Stopwatch sample_timer;
206  util::Stopwatch maintenance_timer;
207  maintenance_timer.start();
208  long checkpoint_counter = 0;
209 
210  TimeSnap timesnap;
211  CPUSnap cpusnap;
212  IOSnap iosnap;
213  SchedSnap schedsnap;
214  NetSnap netsnap;
215  VMSnap vmsnap;
216  ProcSnap procsnap;
217  ResSnap ressnap;
218  MountSnap mountsnap;
219  TCPEstaSnap tcpestasnap;
220  timesnap.startSnap();
221  cpusnap.startSnap();
222  iosnap.startSnap();
223  schedsnap.startSnap();
224  netsnap.startSnap();
225  vmsnap.startSnap();
226  procsnap.startSnap();
227  ressnap.startSnap();
228  mountsnap.startSnap();
229  tcpestasnap.startSnap();
230 
231  while ( !stopped ) {
232 
233  sample_timer.start();
234 
235  if ( maintenance_timer.getElapsedSeconds() > maintenance_interval ) {
236  sysLog( LOG_DEBUG, util::ConfigFile::getConfig()->getIntValue("LOG_LEVEL"), "maintenance: cleanup, checkpoint, vacuum, analyze" );
237  db.begin();
238  applyRetention( db );
239  db.commit();
240  shrinkDB(db, options.database );
241  vacuumAnalyze( db );
242  db.checkPointPassive();
243  std::stringstream ss;
244  ss << "sqlite memory used: " << util::ByteStr( persist::Database::memUsed(), 2 );
245  ss << ", highwater: " << util::ByteStr( persist::Database::memHighWater(), 2 );
246  ss << ", heap softlimit: " << util::ByteStr( persist::Database::softHeapLimit(-1), 2 );
247  sysLog( LOG_DEBUG, util::ConfigFile::getConfig()->getIntValue("LOG_LEVEL"), ss.str() );
248  maintenance_timer.start();
249  }
250 
251  while ( sample_timer.getElapsedSeconds() < sample_interval && !stopped ) {
252  leanux::util::Sleep( 0, 400 * 1000 * 1000 );
253  }
254 
255  timesnap.stopSnap();
256 
257  if ( timesnap.getSeconds() > 0.0 ) {
258  db.begin();
259  long snapid = timesnap.storeSnap( db, 0, 0 );
260  double timesnap_seconds = timesnap.getSeconds();
261  timesnap.startSnap();
262 
263  schedsnap.stopSnap();
264  schedsnap.storeSnap( db, snapid, timesnap_seconds );
265  schedsnap.startSnap();
266 
267  netsnap.stopSnap();
268  netsnap.storeSnap( db, snapid, timesnap_seconds );
269  netsnap.startSnap();
270 
271  vmsnap.stopSnap();
272  vmsnap.storeSnap( db, snapid, timesnap_seconds );
273  vmsnap.startSnap();
274 
275  ressnap.stopSnap();
276  ressnap.storeSnap( db, snapid, timesnap_seconds );
277  ressnap.startSnap();
278 
279  mountsnap.stopSnap();
280  mountsnap.storeSnap( db, snapid, timesnap_seconds );
281  mountsnap.startSnap();
282 
283  tcpestasnap.stopSnap();
284  tcpestasnap.storeSnap( db, snapid, timesnap_seconds );
285  tcpestasnap.startSnap();
286 
287  procsnap.stopSnap();
288  procsnap.storeSnap( db, snapid, timesnap_seconds );
289  procsnap.startSnap();
290 
291  iosnap.stopSnap();
292  iosnap.storeSnap( db, snapid, timesnap_seconds );
293  iosnap.startSnap();
294 
295  cpusnap.stopSnap();
296  cpusnap.storeSnap( db, snapid, timesnap_seconds );
297  cpusnap.startSnap();
298 
299  db.commit();
300  checkpoint_counter++;
301  std::stringstream ss;
302  ss << "committed snapshot " << snapid;
303  sysLog( LOG_DEBUG, util::ConfigFile::getConfig()->getIntValue("LOG_LEVEL"), ss.str() );
304  }
305  if ( was_busy ) {
306  sysLog( LOG_STAT, util::ConfigFile::getConfig()->getIntValue("LOG_LEVEL"), "sampling resumed" );
307  was_busy = false;
308  }
309 
310  if ( checkpoint_counter >= snapshot_checkpoint ) {
311  sysLog( LOG_DEBUG, util::ConfigFile::getConfig()->getIntValue("LOG_LEVEL"), "issuing checkpoint" );
312  db.checkPointPassive();
313  checkpoint_counter = 0;
314  }
315  }
316  try {
317  persist::DML dml(db);
318  dml.prepare( "DELETE FROM status WHERE name=:name" );
319  dml.bind( 1, pid_name );
320  dml.execute();
321  }
322  catch ( const Oops &oops ) {
323  }
324  db.checkPointTruncate();
325  sysLog( LOG_STAT, util::ConfigFile::getConfig()->getIntValue("LOG_LEVEL"), "stopped" );
326  }

328  bool parseArgs( int argc, char* argv[] ) {
329  options.database = "";
330  options.config = "";
331  int opt;
332  while ( (opt = getopt( argc, argv, "f:c:d" ) ) != -1 ) {
333  switch ( opt ) {
334  case 'f':
335  options.database = optarg;
336  break;
337  case 'c':
338  options.config = optarg;
339  break;
340  case 'd':
341  options.daemonize = true;
342  break;
343  default:
344  return false;
345  };
346  }
347  if ( options.database == "" && options.config == "" ) {
348  std::cerr << "specify both -f databasefile -c configfile" << std::endl;
349  return false;
350  }
351  return true;
352  }
