59  void ConfigFile::read() {
60  std::ifstream ifs( filename_.c_str() );
61  while( ifs.good() ) {
62  std::string line;
63  getline( ifs, line );
64  if ( ifs.good() ) {
65  unsigned int state = 0;
66  const unsigned int state_line_start = 0;
67  const unsigned int state_line_type = 1;
68  const unsigned int state_read_name = 2;
69  const unsigned int state_trimws = 3;
70  const unsigned int state_read_value = 4;
71  const unsigned int state_finish = 100;
72  std::string name = "";
73  std::string value = "";
74  size_t pos = 0;
75  while ( state != state_finish && pos < line.length() ) {
76  switch ( state ) {
77  case state_line_start: //initial for each line
78  if ( !std::isspace( line[pos] ) ) state = state_line_type; else pos++;
79  break;
80  case state_line_type: //what sort of line?
81  if ( line[pos] == '#' ) state = state_finish;
82  else if ( !std::isspace(line[pos]) ) {
83  state = state_read_name;
84  }
85  break;
86  case state_read_name: //read the name
87  if ( line[pos] == '=' ) {
88  state = state_trimws;
89  } else if ( !std::isspace(line[pos]) ) {
90  name.push_back( line[pos] );
91  }
92  pos++;
93  break;
94  case state_trimws: //trim ws before value
95  if ( !std::isspace(line[pos]) ) state = state_read_value; else pos++;
96  break;
97  case state_read_value: //read the value
98  if ( pos < line.length() ) {
99  value+= line[pos++];
100  if ( pos >= line.length() ) state = state_finish;
101  } else state = state_finish;
102  break;
103  }
104  }
105  if ( name != "" && value != "" ) {
106  std::map<std::string,Parameter>::const_iterator def = defaults_.find( name );
107  if ( def == defaults_.end() ) throw Oops( __FILE__, __LINE__, "invalid configuration parameter '" + name + "'" );
108  Parameter param( name, value );
109  values_[name] = param;
110  }
111  }
112  }
113  }

115  void ConfigFile::write( bool defaults) {
116  std::ofstream ofs( filename_.c_str() );
117  if ( ofs.good() ) {
118  ofs << "# " << appname_ << " configuration file" << std::endl << std::endl;
119  //ofs << "# specify RGB values as (RED,GREEN,BLUE), each scalar >=0 and <=255" << std::endl << std::endl;
120  for ( std::map<std::string,Parameter>::const_iterator d = defaults_.begin(); d != defaults_.end(); ++d ) {
121  std::string write_value;
122  if ( defaults ) {
123  write_value = d->second.value_;
124  } else {
125  std::map<std::string,Parameter>::const_iterator user = values_.find( d->first );
126  if ( user == values_.end() ) write_value = d->second.value_; else write_value = user->second.value_;
127  }
128  ofs << "# " << d->first << ": " << descriptions_[d->first] << std::endl;
129  //
130  std::list<std::string> tokens;
131  std::set<std::string> omit;
132  util::tokenize( comments_[d->first], tokens, ' ', omit );
133  int cr = 2;
134  if ( tokens.size() > 0 ) {
135  ofs << "# ";
136  for ( std::list<std::string>::const_iterator t = tokens.begin(); t != tokens.end(); t++ ) {
137  if ( cr + (*t).length() > 80 ) {
138  ofs << std::endl << "# " << *t << " ";
139  cr = 2;
140  } else {
141  ofs << *t << " ";
142  cr += (*t).length();
143  }
144  }
145  ofs << std::endl;
146  }
147  //
148  ofs << "# default " << d->first << "=" << d->second.value_ << std::endl;
149  ofs << d->first << "=" << write_value << std::endl;
150  ofs << std::endl;
151  }
152  } else throw Oops( __FILE__, __LINE__, "unable to write to '" + filename_ + "'" );
153  }

172  ConfigFile::RGB ConfigFile::getRGBValue( const std::string &name ) const {
173  ConfigFile::RGB rgb;
174  std::string s = getValue( name );
175  const unsigned int state_initial = 0;
176  const unsigned int state_red = 1;
177  const unsigned int state_green = 2;
178  const unsigned int state_blue = 3;
179  const unsigned int state_finished = 100;
180  const unsigned int state_invalid = 101;
181  unsigned int state = state_initial;
182  std::string red = "";
183  std::string green = "";
184  std::string blue = "";
185  unsigned int pos = 0;
186  while ( pos < s.length() ) {
187  switch ( state ) {
188  case state_initial:
189  if ( s[pos] == '(' ) state = state_red; else state = state_invalid;
190  pos++;
191  break;
192  case state_red:
193  if ( s[pos] != ',' ) {
194  red += s[pos];
195  } else state = state_green;
196  pos++;
197  break;
198  case state_green:
199  if ( s[pos] != ',' ) {
200  green += s[pos];
201  } else state = state_blue;
202  pos++;
203  break;
204  case state_blue:
205  if ( s[pos] != ')' ) {
206  blue += s[pos];
207  } else state = state_finished;
208  pos++;
209  break;
210  }
211  }
212  if ( state != state_finished ) throw Oops( __FILE__, __LINE__, "invalid RGB value for '" + name + "'" );
213  else {
214  rgb.red = atoi( red.c_str() );
215  rgb.green = atoi( green.c_str() );
216  rgb.blue = atoi( blue.c_str() );
217  }
218  return rgb;
219  }
