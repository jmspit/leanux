73  tokens.clear();

174  std::string ByteStr( double bytes, int prec ) {
175  char bytestrbuf[bytestr_max_width+1];
176  bytestrbuf[0] = 0;
177  char format[20];
178  std::string suffix;
179  if ( isnan( bytes ) ) {
180  strncat( bytestrbuf, "nan", bytestr_max_width );
181  } else
182  if ( bytes != 0.0 ) {
183  // the order of the number
184  double baseorder = floor( log( fabs(bytes) )/base_1024_log );
185  // the scale multiplier
186  double scale = floor( baseorder / 3.0 );
187  // the order of the number remaining after eating away scale powers of 1024^(1/3)
188  double remainorder = floor( log(bytes/pow(base_1024,scale*3))/log(10) );
189  double p = 0;
190  if ( remainorder == 0 ) p = prec-1;
191  else if ( remainorder == 1 ) p = prec-2;
192  else if ( remainorder == 2 ) p = prec-3;
193  else if ( remainorder == 3 ) p = 0;
194  if ( p < 0 )
195  snprintf( format, 20, "%%.0f%%s" );
196  else
197  snprintf( format, 20, "%%.%.0ff%%s", p );
198 
199  if ( scale < 0 ) {
200  suffix = "";
201  if ( prec-1 < 0 )
202  snprintf( format, 20, "%%.0%%s" );
203  else
204  snprintf( format, 20, "%%.%if%%s", prec-1 );
205  scale = 0;
206  }
207  else if ( scale < 1 ) suffix = "";
208  else if ( scale < 2 ) suffix = "K";
209  else if ( scale < 3 ) suffix = "M";
210  else if ( scale < 4 ) suffix = "G";
211  else if ( scale < 5 ) suffix = "T";
212  else if ( scale < 6 ) suffix = "P";
213  else if ( scale < 7 ) suffix = "E";
214  else if ( scale < 8 ) suffix = "Z";
215  else if ( scale < 9 ) suffix = "Y";
216  snprintf( bytestrbuf, bytestr_max_width, format, bytes/pow(base_1024,scale*3.0), suffix.c_str() );
217  } else strncat( bytestrbuf, "0", bytestr_max_width );
218  return bytestrbuf;
219  }

225  std::string TimeStrMicro( double time ) {
226  char buffer[128];
227  char mu = 'u';
228  if ( fabs(time) == 0.0 )
229  return "0s";
230  else if ( fabs(time) < 1000.0 )
231  snprintf( buffer, 128, "%0.f%cs", time, mu );
232  else if ( fabs(time) < 100000.0 )
233  snprintf( buffer, 128, "%.1fms", time / 1000.0 );
234  else if ( fabs(time) < 1000000.0 )
235  snprintf( buffer, 128, "%.0fms", time / 1000.0 );
236  else if ( fabs(time) < 1000000.0*60.0 )
237  snprintf( buffer, 128, "%.2fs", time / 1000000.0 );
238  else if ( fabs(time) < 1000000.0*60.0*60.0 )
239  if ( fmod( time, 60.0*1000000.0 )/1000000.0 >= 0.5 )
240  snprintf( buffer, 128, "%0.fm%0.fs", floor( time / (60.0*1000000.0) ), fmod( time, 60.0*1000000.0 )/1000000.0 );
241  else
242  snprintf( buffer, 128, "%0.fm", floor( time / (60.0*1000000.0) ) );
243  else if ( fabs(time) < 1000000.0*60.0*60.0*24.0 )
244  {
245  double h = floor( time / (60.0*60.0*1000000.0) );
246  double m = floor( ( time - h * 60.0*60.0*1000000.0 ) / (60.0*1000000.0) );
247  double s = ( time - h * 60.0*60.0*1000000.0 - m * 60.0*1000000.0 ) / 1000000.0;
248  if ( s >= 0.5 )
249  snprintf( buffer, 128, "%0.fh%0.fm%0.fs", h, m, s );
250  else
251  snprintf( buffer, 128, "%0.fh%0.fm", h, m );
252  } else {
253  double d = floor( time / (24.0*60.0*60.0*1000000.0) );
254  double ts = d * (24.0*60.0*60.0*1000000.0);
255  double h = floor( (time-ts) / (60.0*60.0*1000000.0) );
256  ts += h * (60.0*60.0*1000000.0);
257  double m = floor( ( time - ts ) / (60.0*1000000.0) );
258  ts += m * (60.0*1000000.0);
259  double s = ( time - ts ) / 1000000.0;
260  snprintf( buffer, 128, "%0.fd%0.fh%0.fm%0.fs", d, h, m, s );
261  }
262  return buffer;
263  }

265  std::string NumStr( double num, int prec ) {
266  char numstrbuf[32];
267  numstrbuf[0] = 0;
268  char format[20];
269  std::string suffix;
270  if ( isnan( num ) ) {
271  strncat( numstrbuf, "nan", sizeof(numstrbuf)-1 );
272  } else
273  if ( num != 0.0 ) {
274  // the order of the number
275  double baseorder = floor( log10( fabs(num) ) );
276  // the scale multiplier, 3=k=1E3, 6=M=1E6, 9=G=1E9, 12=T=1E12
277  double scale = floor( baseorder / 3.0 );
278  // the order of the number remaining after eating away scale powers of 10
279  double remainorder = floor( log10(num/pow(10.0,scale*3)) );
280  if ( prec > fabs(remainorder) + 1 )
281  snprintf( format, 20, "%%.%.0ff%%s", prec-(remainorder+1) );
282  else
283  snprintf( format, 20, "%%.0f%%s" );
284 
285  if ( scale < 0 ) {
286  suffix = "";
287  snprintf( format, 20, "%%.%if%%s", prec-1 );
288  scale = 0;
289  }
290  else if ( scale < 1 ) suffix = "";
291  else if ( scale < 2 ) suffix = "k";
292  else if ( scale < 3 ) suffix = "M";
293  else if ( scale < 4 ) suffix = "G";
294  else if ( scale < 5 ) suffix = "T";
295  else if ( scale < 6 ) suffix = "P";
296  else if ( scale < 7 ) suffix = "E";
297  else if ( scale < 8 ) suffix = "Z";
298  snprintf( numstrbuf, sizeof(numstrbuf), format, num/pow(10.0,scale*3.0), suffix.c_str() );
299  } else strncat( numstrbuf, "0", sizeof(numstrbuf)-1 );
300  return numstrbuf;
301  }
