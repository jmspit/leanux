76  void enumUSBDevices( std::list<USBDevicePath> &paths ) {
77  paths.clear();
78  DIR *d;
79  struct dirent *dir;
80  std::string resolved_path;
81  d = opendir( (sysdevice::sysbus_root + "/usb/devices").c_str() );
82  if ( d ) {
83  while ( (dir = readdir(d)) != NULL ) {
84  if ( dir->d_name[0] != '.' && ! strchr( dir->d_name, ':' ) ) {
85  std::string path = sysdevice::sysbus_root + "/usb/devices/" + (std::string)dir->d_name;
86  resolved_path = util::realPath( path );
87  paths.push_back( resolved_path );
88  }
89  }
90  closedir( d );
91  }
92  }

144  bool getUSBHardwareInfo( const USBHardwareId &id, USBHardwareInfo &inf ) {
145  inf.idVendor = "";
146  inf.idProduct = "";
147  std::stringstream ss;
148  ss << std::hex;
149  ss << std::setfill('0') << std::setw(4) << id.idVendor;
150  std::string hex_vendor = ss.str();
151  ss.str("");
152  ss << std::setfill('0') << std::setw(4) << id.idProduct;
153  std::string hex_product = ss.str();
154  inf.idProduct = "unknown product id 0x" + hex_product;
155 
156  std::ifstream f( USBDeviceDatabase.c_str() );
157  if ( !f.good() ) throw Oops( __FILE__, __LINE__, "error opening usb.ids" );
158 
159  int state = 0;
160 
161  bool giveup = false;
162  while ( f.good() && !f.eof() && !giveup && state != 2 ) {
163  std::string s = "";
164  getline( f, s );
165 
166  if ( s[0] != '#' ) {
167 
168  switch ( state ) {
169  case 0:
170  if ( strncmp( s.c_str(), hex_vendor.c_str(), 4 ) == 0 ) {
171  inf.idVendor = s.substr( 6 );
172  state = 1;
173  }
174  break;
175  case 1:
176  if ( s[0] != '\t' ) {
177  //we hit another vendor line, so device not found
178  giveup = true;
179  } else {
180  if ( strncmp( s.c_str()+1, hex_product.c_str(), 4 ) == 0 ) {
181  inf.idProduct = s.substr(7);
182  state = 2;
183  }
184  }
185  break;
186  }
187  }
188  }
189  return ( state >= 1 );
190  }
