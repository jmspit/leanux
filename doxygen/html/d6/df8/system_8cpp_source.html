66  void init() {
67  std::string version = system::getKernelVersion();
68  if ( version < kernel_required )
69  throw Oops( __FILE__, __LINE__, "leanux requires at least " + kernel_required + ", this kernel is " + version );
70  if ( !util::directoryExists( "/proc" ) ) throw Oops( __FILE__, __LINE__, "leanux requires /proc procfs" );
71  if ( !util::directoryExists( "/sys" ) ) throw Oops( __FILE__, __LINE__, "leanux requires /sys sysfs" );
72  pci::init();
73  block::init();
74  net::init();
75  usb::init();
76  }

173  time_t getBootTime() {
174  std::ifstream ifs( "/proc/stat" );
175  double t = 0;
176  while ( ifs.good() && !ifs.eof() ) {
177  std::string s;
178  getline( ifs, s );
179  if ( sscanf( s.c_str(), "btime %lf", &t ) == 1 ) return (time_t)t; else t= 0;
180  }
181  throw Oops( __FILE__, __LINE__, "no btime in /proc/stat" );
182  return (time_t)t;
183  }

393  Distribution getDistribution() {
394  Distribution dist;
395  dist.type = Unkown;
396  dist.release = "";
397  // test for os-release
398  if ( util::fileReadAccess( "/etc/os-release" ) ) {
399  std::string distrib_id = readOSReleaseDistribId();
400  dist.release = readOSReleaseName();
401  if ( readOSReleaseVersion().length() > 0 ) dist.release = dist.release + " " + readOSReleaseVersion();
402  if ( distrib_id == "gentoo" ) {
403  dist.type = Gentoo;
404  } else if ( distrib_id == "opensuse" ) {
405  dist.type = OpenSuSE;
406  } else if ( distrib_id == "centos" ) {
407  dist.type = CentOS;
408  } else if ( distrib_id == "ubuntu" ) {
409  dist.type = Ubuntu;
410  } else if ( distrib_id == "debian" ) {
411  dist.type = Debian;
412  }
413  } else
414  // test for lsb-release
415  if ( util::fileReadAccess( "/etc/lsb-release" ) ) {
416  std::string distrib_id = readLSBReleaseDistribId();
417  if ( distrib_id == "Gentoo" ) {
418  dist.type = Gentoo;
419  dist.release = util::fileReadString( "/etc/gentoo-release" );
420  } else if ( distrib_id == "Ubuntu" ) {
421  dist.type = Ubuntu;
422  dist.release = readLSBReleaseDistribDescription();
423  } else if ( distrib_id == "centos" ) {
424  dist.type = Ubuntu;
425  dist.release = readLSBReleaseDistribDescription();
426  }
427  } else {
428  // no lsb-release, alternative detection
429  if ( util::fileReadAccess( "/etc/gentoo-release" ) ) {
430  dist.type = Gentoo;
431  dist.release = util::fileReadString( "/etc/gentoo-release" );
432  } else if ( util::fileReadAccess( "/etc/debian-version" ) ) {
433  dist.type = Debian;
434  dist.release = util::fileReadString( "/etc/debian-version" );
435  } else if ( util::fileReadAccess( "/etc/redhat-release" ) ) {
436  dist.type = RedHat;
437  dist.release = util::fileReadString( "/etc/redhat-release" );
438  } else if ( util::fileReadAccess( "/etc/SuSE-release" ) ) {
439  dist.type = OpenSuSE;
440  dist.release = util::fileReadString( "/etc/SuSE-release" );
441  }
442  }
443  return dist;
444  }
