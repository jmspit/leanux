129  bool getOptions( int argc, char* argv[] ) {
130  options.opt_v = false;
131  options.opt_l = false;
132  options.opt_t = "";
133  options.device = "";
134  int opt;
135  while ( (opt = getopt( argc, argv, "hvlt:f:" ) ) != -1 ) {
136  switch ( opt ) {
137  case 'f':
138  options.opt_f = optarg;
139  break;
140  case 't':
141  options.opt_t = optarg;
142  break;
143  case 'v':
144  options.opt_v = true;
145  break;
146  case 'h':
147  options.opt_h = true;
148  break;
149  case 'l':
150  options.opt_l = true;
151  break;
152  default:
153  return false;
154  };
155  }
156  if ( optind == argc -1 ) {
157  options.device = argv[optind];
158  } else if ( optind < argc -1 ) return false;
159 
160  // the user specified type selector must exist
161  if ( options.opt_t != "" ) {
162  TypeMap::const_iterator tm = type_map.find( options.opt_t );
163  if ( tm == type_map.end() ) {
164  std::cerr << "invalid type '" << options.opt_t << "'" << std::endl;
165  return false;
166  }
167  }
168 
169  // the user specified display field must exist
170  if ( options.opt_f != "" ) {
171  // but -f requires -t
172  if ( options.opt_t != "" ) {
173  std::set<std::string> empty;
174  DisplayFields temp_fields;
175  util::tokenize( options.opt_f, temp_fields, ',', empty );
176  for ( DisplayFields::const_iterator dc = temp_fields.begin(); dc != temp_fields.end(); dc++ ) {
177  std::string field = "";
178  bool hidden = false;
179  Filter temp_filter;
180  size_t p = (*dc).find_first_not_of("abcdefghijklmnopqrstuvwxyz_0123456789*");
181  if ( p != std::string::npos ) {
182  if ( (*dc)[p] != '=' && (*dc)[p] != '-' ) {
183  std::cerr << "invalid field specification '" << *dc << "'" << std::endl;
184  return false;
185  }
186  temp_filter.negate = (*dc)[p] == '-';
187  temp_filter.filter = (*dc).substr(p+1);
188  field = (*dc).substr(0,p);
189  } else {
190  field = *dc;
191  temp_filter.filter = "";
192  temp_filter.negate = false;
193  }
194  if ( field.length()> 0 && field[0] == '*' ) {
195  hidden = true;
196  field = field.substr(1);
197  }
198  TypeFields::const_iterator found = type_fields.find(field);
199  if ( found != type_fields.end() ) {
200  if (!hidden ) display_fields.push_back(field);
201  if ( temp_filter.filter != "" ) {
202  filter_map[field] = temp_filter;
203  }
204  } else {
205  std::cerr << "invalid field '" << field << "'" << std::endl;
206  return false;
207  }
208  }
209  } else {
210  std::cerr << "-f requires -t" << std::endl;
211  return false;
212  }
213  }
214 
215  return true;
216  };

282  void sysfsTree( const std::string &sysfs) {
283  tools::Tabular tab;
284  tab.addColumn( "sysfs device", false );
285  tab.addColumn( "type", false );
286  tab.addColumn( "driver", false );
287  tab.addColumn( "class", false );
288  tab.addColumn( "attributes", false );
289  std::list<leanux::sysdevice::SysDevice*> devices;
290  size_t path_indent = 0;
291  if ( leanux::sysdevice::treeDetect( sysfs, devices ) ) {
292  if ( !devices.size() ) std::cerr << "0 devices found" << std::endl;
293  unsigned int depth = 0;
294  for ( std::list<leanux::sysdevice::SysDevice*>::const_iterator d = devices.begin(); d != devices.end(); d++, depth++ ) {
295  tab.appendString( "sysfs device", std::string(depth,' ') + util::shortenString( (*d)->getPath().substr(path_indent), 160, '_' ) );
296  tab.appendString( "type", leanux::sysdevice::SysDevice::getTypeStr( (*d)->getType() ) );
297  tab.appendString( "driver", (*d)->getDriver() );
298  tab.appendString( "class", (*d)->getClass() );
299  tab.appendString( "attributes", (*d)->getDescription() );
300  path_indent += ((*d)->getPath().length() - path_indent) + 1;
301  }
302  tab.dump(std::cout);
303  leanux::sysdevice::destroy( devices );
304  } else std::cerr << "detection error" << std::endl;
305  }

310  std::string getLeafColStr( const sysdevice::SysDevice* leaf, const std::string &field ) {
311  if ( field == "dev_t" ) {
312  const sysdevice::BlockDevice* block_device = dynamic_cast<const sysdevice::BlockDevice*>(leaf);
313  if ( block_device ) {
314  std::stringstream ss;
315  ss << block_device->getMajorMinor();
316  return ss.str();
317  }
318  } else if ( field == "class" ) {
319  const sysdevice::PCIDevice* pci_device = dynamic_cast<const sysdevice::PCIDevice*>(leaf);
320  if ( pci_device ) {
321  pci::PCIHardwareId hwid = pci::getPCIHardwareId( leaf->getPath() );
322  pci::PCIHardwareInfo hwinfo;
323  if ( pci::getPCIHardwareInfo( hwid, hwinfo ) ) {
324  return hwinfo.pciclass;
325  }
326  }
327  const sysdevice::USBInterface* usb_iface = dynamic_cast<const sysdevice::USBInterface*>(leaf);
328  if ( usb_iface ) {
329  return usb::getUSBHardwareClassStr( usb::getUSBInterfaceHardwareClass( leaf->getPath() ) );
330  }
331  const sysdevice::USBBus* usb_bus = dynamic_cast<const sysdevice::USBBus*>(leaf);
332  if ( usb_bus ) {
333  return usb::getUSBHardwareClassStr( usb::getUSBDeviceHardwareClass( leaf->getPath() ) );
334  }
335  const sysdevice::USBDevice* usb_device = dynamic_cast<const sysdevice::USBDevice*>(leaf);
336  if ( usb_device ) {
337  return usb::getUSBHardwareClassStr( usb::getUSBDeviceHardwareClass( leaf->getPath() ) );
338  }
339  } else if ( field == "driver" ) {
340  return leaf->getDriver();
341  } else if ( field == "sysfs" ) {
342  return leaf->getPath();
343  } else if ( field == "ip4" ) {
344  const sysdevice::NetDevice* net_device = dynamic_cast<const sysdevice::NetDevice*>(leaf);
345  if ( net_device ) {
346  std::list<std::string> ip4;
347  net::getDeviceIP4Addresses( util::leafDir( net_device->getPath() ), ip4 );
348  std::stringstream ss;
349  for ( std::list<std::string>::const_iterator i = ip4.begin(); i != ip4.end(); i ++ ) {
350  if ( i != ip4.begin() ) ss << ", ";
351  ss << *i;
352  }
353  return ss.str();
354  }
355  } else if ( field == "ip6" ) {
356  const sysdevice::NetDevice* net_device = dynamic_cast<const sysdevice::NetDevice*>(leaf);

365  return ss.str();
366  }
367  } else if ( field == "iqn" ) {
368  const sysdevice::iSCSIDevice* iscsi_device = dynamic_cast<const sysdevice::iSCSIDevice*>(leaf);
369  if ( iscsi_device ) {
370  return iscsi_device->getTargetIQN();

372  } else if ( field == "irq" ) {
373  const sysdevice::PCIBus* pci_bus = dynamic_cast<const sysdevice::PCIBus*>(leaf);
374  if ( pci_bus ) {
375  std::stringstream ss;
376  ss << pci::getPCIDeviceIRQ( pci_bus->getPath() );
377  return ss.str();
378  }
379  const sysdevice::PCIDevice* pci_device = dynamic_cast<const sysdevice::PCIDevice*>(leaf);

381  std::stringstream ss;

385  } else if ( field == "model" ) {
386  const sysdevice::BlockDevice* block_device = dynamic_cast<const sysdevice::BlockDevice*>(leaf);

550  void listDevices( const std::list<sysdevice::SysDevicePath> &devices, sysdevice::SysDevice::SysDeviceType t ) {
551  tools::Tabular tab;
552  tab.addColumn( "sysfs", false );
553  tab.addColumn( "type", false );
554  for ( DisplayFields::const_iterator dc = display_fields.begin(); dc != display_fields.end(); dc++ ) {
555  if ( *dc != "sysfs" && *dc != "type" ) tab.addColumn( *dc, false );
556  }
557  for ( std::list<sysdevice::SysDevicePath>::const_iterator d = devices.begin(); d != devices.end(); d++ ) {
558  std::string parent = "";
559  sysdevice::SysDevice* leaf = sysdevice::leafDetect( *d, parent );
560  if ( leaf ) {
561  if ( options.opt_t == "" || leaf->matchSysDeviceType( t ) ) {
562  std::map<std::string,std::string> temp;
563  bool include = true;
564  for ( FilterMap::const_iterator fm = filter_map.begin(); fm != filter_map.end(); ++fm ) {
565  std::string s = getLeafColStr( leaf, fm->first );
566  util::RegExp reg;
567  reg.set( fm->second.filter );
568  if ( (!fm->second.negate && !reg.match( s ) ) ||
569  (fm->second.negate && reg.match( s ) ) ) {
570  include = false;
571  break;
572  }
573  }
574  if ( include ) {
575  temp["sysfs"] = leaf->getPath();
576  temp["type"] = leanux::sysdevice::SysDevice::getTypeStr( leaf->getType() );
577  for ( DisplayFields::const_iterator dc = display_fields.begin(); dc != display_fields.end(); dc++ ) {
578  temp[*dc] = getLeafColStr( leaf, *dc );
579  }
580  for ( std::map<std::string,std::string>::const_iterator t = temp.begin(); t != temp.end(); ++t ) {
581  tab.appendString( t->first, t->second );
582  }
583  }
584  }
585  delete leaf;
586  }
587  }
588  tab.dump(std::cout);
589  }
