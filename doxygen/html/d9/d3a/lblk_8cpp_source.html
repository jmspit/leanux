133  bool getOptions( int argc, char* argv[] ) {
134  options.opt_v = false;
135  options.opt_d = false;
136  options.opt_m = false;
137  options.opt_h = false;
138  options.opt_f = false;
139  options.opt_l = false;
140  options.opt_w = "";
141  options.device = "";
142  int opt;
143  while ( (opt = getopt( argc, argv, "vlfthmdw:" ) ) != -1 ) {
144  switch ( opt ) {
145  case 'd':
146  options.opt_d = true;
147  break;
148  case 'w':
149  options.opt_w = optarg;
150  break;
151  case 'm':
152  options.opt_m = true;
153  break;
154  case 'h':
155  options.opt_h = true;
156  break;
157  case 't':
158  options.opt_t = true;
159  break;
160  case 'f':
161  options.opt_f = true;
162  break;
163  case 'l':
164  options.opt_l = true;
165  break;
166  case 'v':
167  options.opt_v = true;
168  break;
169  default:
170  return false;
171  };
172  }
173  if ( optind == argc -1 ) {
174  options.device = argv[optind];
175  } else if ( optind < argc -1 ) return false;
176  if ( options.opt_w.length() > 0 && ! options.opt_d ) {
177  std::cerr << "cannot use -w without -d" << std::endl;
178  return false;
179  }
180  if ( options.opt_t && options.device == "" ) {
181  std::cerr << "-t requires a device argument" << std::endl;
182  return false;
183  }
184  if ( options.opt_f && options.device == "" ) {
185  std::cerr << "-f requires a file or filesystem argument" << std::endl;
186  return false;
187  }
188  return true;
189  };

197  void printHolderTree( const block::MajorMinor &mm, Tabular& tab, unsigned int level = 0 ) {
198  if ( tab.columnCount() == 0 ) {
199  tab.addColumn( "holder tree", false );
200  tab.addColumn( "dev", false );
201  tab.addColumn( "class", false );
202  tab.addColumn( "size" );
203  tab.addColumn( "svct" );
204  tab.addColumn( "r/(r+w)" );
205  tab.addColumn( "description", false );
206  }
207  std::list<std::string> holders;
208  std::list<std::string> parts;
209  std::string devname = block::MajorMinor::getNameByMajorMinor( mm );
210  std::stringstream ss;
211  if ( level > 0 ) ss << std::setfill(' ') << std::setw(level*2) << " " << std::setfill(' ');
212  ss << mm.getName();
213  tab.appendString( "holder tree", ss.str() );
214  ss.str("");
215  ss << mm;
216  tab.appendString( "dev", ss.str() );
217  tab.appendString( "class", mm.getClassStr() );
218  tab.appendString( "size", util::ByteStr( mm.getSize(), 3 ) );
219 
220  block::DeviceStats stats;
221  if ( mm.getStats( stats ) ) {
222  if ( stats.reads+stats.writes > 0 ) {
223  tab.appendString( "svct", util::TimeStrSec( (stats.io_ms/1000.0) / (double)(stats.reads+stats.writes) ) );
224  tab.appendString( "r/(r+w)", util::NumStr( (double)stats.reads/(double)(stats.reads+stats.writes) ) );
225  } else {
226  tab.appendString( "svct", " " );
227  tab.appendString( "r/(r+w)", " " );
228  }
229  } else {
230  tab.appendString( "svct", " " );
231  tab.appendString( "r/(r+w)", " " );
232  }
233 
234  tab.appendString( "description", mm.getDescription() );
235 
236  mm.getPartitions( parts );
237  mm.getHolders( holders );
238  for ( std::list<std::string>::const_iterator d = holders.begin(); d != holders.end(); d++ ) {
239  printHolderTree( block::MajorMinor::getMajorMinorByName( (*d) ), tab, level+1 );
240  }
241  for ( std::list<std::string>::const_iterator d = parts.begin(); d != parts.end(); d++ ) {
242  printHolderTree( block::MajorMinor::getMajorMinorByName( (*d) ), tab, level+1 );
243  }
244  }

252  void printSlaveTree( const block::MajorMinor &mm, Tabular& tab, unsigned int level = 0 ) {
253  if ( tab.columnCount() == 0 ) {
254  tab.addColumn( "slave tree", false );
255  tab.addColumn( "dev", false );
256  tab.addColumn( "class", false );
257  tab.addColumn( "size" );
258  tab.addColumn( "svct" );
259  tab.addColumn( "r/(r+w)" );
260  tab.addColumn( "description", false );
261  }
262  std::list<std::string> slaves;
263  std::string devname = block::MajorMinor::getNameByMajorMinor( mm );
264  std::stringstream ss;
265  if ( level > 0 ) ss << std::setfill(' ') << std::setw(level*2) << " " << std::setfill(' ');
266  ss << mm.getName();
267  tab.appendString( "slave tree", ss.str() );
268  ss.str("");
269  ss << mm;
270  tab.appendString( "dev", ss.str() );
271  tab.appendString( "class", mm.getClassStr() );
272  tab.appendString( "size", util::ByteStr( mm.getSize(), 3 ) );
273 
274  block::DeviceStats stats;
275  if ( mm.getStats( stats ) ) {
276  if ( stats.reads+stats.writes > 0 ) {
277  tab.appendString( "svct", util::TimeStrSec( (stats.io_ms/1000.0) / (double)(stats.reads+stats.writes) ) );
278  tab.appendString( "r/(r+w)", util::NumStr( (double)stats.reads/(double)(stats.reads+stats.writes) ) );
279  } else {
280  tab.appendString( "svct", " " );
281  tab.appendString( "r/(r+w)", " " );
282  }
283  } else {
284  tab.appendString( "svct", " " );
285  tab.appendString( "r/(r+w)", " " );
286  }
287 
288  tab.appendString( "description", mm.getDescription() );
289 
290  mm.getSlaves( slaves );
291  for ( std::list<std::string>::const_iterator d = slaves.begin(); d != slaves.end(); d++ ) {
292  printSlaveTree( block::MajorMinor::getMajorMinorByName( (*d) ), tab, level+1 );
293  }
294  if ( mm.isPartition() ) {
295  block::MajorMinor wd = block::MajorMinor::deriveWholeDisk( mm );
296  if ( ! (wd == mm ) ) {
297  printSlaveTree( wd, tab, level+1 );
298  }
299  }

307  void listLVM( std::ostream& os ) {
308  std::list<block::MajorMinor> pvs;
309  block::enumLVMPVS( pvs );
310  if ( pvs.size() > 0 ) {
311  Tabular tab;
312  for ( std::list<block::MajorMinor>::const_iterator p = pvs.begin(); p != pvs.end(); p++ ) {

322  } else {

332  void listFile( std::ostream& os, const std::string &file ) {
333  struct stat st;
334  if ( !stat( file.c_str(), &st ) ) {
335  block::MajorMinor mm = block::MajorMinor( st.st_dev );
336  if ( !mm.isValid() ) mm = block::MajorMinor( st.st_rdev );
337  if ( mm.isValid() && block::MajorMinor::getNameByMajorMinor(mm) != "" ) {
338  Tabular tab;
339  printSlaveTree( mm, tab );
340  tab.dump( os );
341  } else throw Oops( __FILE__, __LINE__, "filesystem stat returns invalid block device" );
342  } else throw Oops( __FILE__, __LINE__, "file not found" );
343  }

527  ss.str("");
528  if ( mm.getRotational() ) {
529  if ( mm.getRPM() > 0 )
530  ss << "spindle (" << mm.getRPM() << "RPM)";
531  else
532  ss << "virtual";

535  tab.appendString( "value", "SSD" );

537

539  tab.appendString( "value", mm.getKernelModule() );
540 
541  std::string hctl = mm.getSCSIHCTL();
542  if ( hctl != "" ) {
543  tab.appendString( "property", "SCSI address" );
544  tab.appendString( "value", hctl );
545  }
546  std::string cache_type = mm.getCacheMode();
547  if ( cache_type != "" ) {
548  tab.appendString( "property", "cache type" );
549  tab.appendString( "value", cache_type );

552  tab.appendString( "property", "IO scheduler" );

565  tab.appendString( "value", util::ByteStr( mm.getReadAhead(), 3 ) );
566 
567  tab.appendString( "property", "SCSI requests" );
568  tab.appendString( "value", util::NumStr( mm.getSCSIIORequest(), 3 ) );
569 
570  tab.appendString( "property", "SCSI iodone" );
571  tab.appendString( "value", util::NumStr( mm.getSCSIIODone(), 3 ) );
572 
573  tab.appendString( "property", "SCSI errors" );
574  std::stringstream ss;
575  ss << util::NumStr( mm.getSCSIIOError(), 3 ) << " (" << util::NumStr( (double)mm.getSCSIIOError()/(double)mm.getSCSIIORequest()*100.0, 3 ) << "%)";
576  tab.appendString( "value", ss.str() );
577 
578  }
579 
580  tab.appendString( "property", "fs type" );
581  tab.appendString( "value", mm.getFSType() );
582 
583  tab.appendString( "property", "fs use" );
584  tab.appendString( "value", mm.getFSUsage() );
585 
586  std::list<std::string> aliases;
587  mm.getAliases( aliases );
588  for ( std::list<std::string>::const_iterator alias = aliases.begin(); alias != aliases.end(); alias++ ) {
589  tab.appendString( "property", "alias" );
590  tab.appendString( "value", *alias );
591  }
592 
593  block::DeviceStats stats;
594  if ( mm.getStats( stats ) ) {
595  if ( stats.reads+stats.writes > 0 ) {
596  tab.appendString( "property", "service time" );
597  tab.appendString( "value", util::TimeStrSec( (stats.io_ms/1000.0) / (double)(stats.reads+stats.writes) ) );
598  tab.appendString( "property", "r/(r+w) ratio" );
599  tab.appendString( "value", util::NumStr( (double)stats.reads/(double)(stats.reads+stats.writes) ) );
600  }
601  }
602 
603  tab.dump( os );
604 
605  tab.clear();
606 
607  os << std::endl;
608  std::list<leanux::sysdevice::SysDevice*> devices;
609  size_t path_indent = 0;
610  block::MajorMinor wholedisk = mm;
611  if ( mm.isPartition() ) wholedisk = block::MajorMinor::deriveWholeDisk( mm );
612  if ( leanux::sysdevice::treeDetect( wholedisk.getSysPath(), devices ) ) {
613  tab.addColumn( "sysfs device", false );
614  tab.addColumn( "type", false );
615  tab.addColumn( "driver", false );
616  tab.addColumn( "class", false );
617  tab.addColumn( "descr", false );
618  unsigned int depth = 0;
619  for ( std::list<leanux::sysdevice::SysDevice*>::const_iterator d = devices.begin(); d != devices.end(); d++, depth++ ) {
620  tab.appendString( "sysfs device", std::string(depth,' ') + (*d)->getPath().substr(path_indent) );
621  tab.appendString( "type", leanux::sysdevice::SysDevice::getTypeStr( (*d)->getType() ) );
622  tab.appendString( "driver", (*d)->getDriver() );
623  tab.appendString( "class", (*d)->getClass() );
624  tab.appendString( "descr", (*d)->getDescription() );
625  path_indent += ((*d)->getPath().length() - path_indent) + 1;
626  }
627  tab.dump(os);
628  leanux::sysdevice::destroy( devices );
629  } else std::cerr << "detection error" << std::endl;
630

729  block::MajorMinor parseDeviceArg() {
730  // is it just a device name?
731  block::MajorMinor mm = block::MajorMinor::getMajorMinorByName( options.device );
732  if ( !mm.isValid() ) {
733  //is it a device path?
734  std::string resolved_path = util::realPath( options.device );
735  struct stat st;
736  if ( !stat( resolved_path.c_str(), &st ) ) {
737  return block::MajorMinor( MAJOR(st.st_rdev), MINOR(st.st_rdev) );
738  }
739  //maybe it is a mapper device?
740  std::stringstream ss;
741  ss << "/dev/mapper/" << options.device;
742  if ( !stat( ss.str().c_str(), &st ) ) {
743  return block::MajorMinor( MAJOR(st.st_rdev), MINOR(st.st_rdev) );
744  }
745  //or a by-id
746  ss.str("");
747  ss << "/dev/disk/by-id/" << options.device;
748  if ( !stat( ss.str().c_str(), &st ) ) {
749  return block::MajorMinor( MAJOR(st.st_rdev), MINOR(st.st_rdev) );
750  }
751  //or a by-uuid
752  ss.str("");
753  ss << "/dev/disk/by-uuid/" << options.device;
754  if ( !stat( ss.str().c_str(), &st ) ) {
755  return block::MajorMinor( MAJOR(st.st_rdev), MINOR(st.st_rdev) );
756  }
757  } else return mm;
758  return block::MajorMinor::invalid;
759  }

764  void runOptions() {
765  if ( !options.opt_v ) {
766  if ( !options.opt_h ) {
767  if ( options.device != "" ) {
768  if ( !options.opt_f ) {
769  block::MajorMinor mm = parseDeviceArg();
770  if ( mm.isValid() ) {
771  if ( options.opt_t ) {
772  Tabular htab;
773  printHolderTree( mm, htab );
774  if ( htab.rowCount() > 2 ) {
775  htab.dump( std::cout );
776  }
777  Tabular stab;
778  printSlaveTree( mm, stab );
779  if ( stab.rowCount() > 2 ) {
780  if ( htab.rowCount() > 2 ) std::cout << std::endl;
781  stab.dump( std::cout );
782  }
783  } else {
784  if ( block::MajorMinor::isSCSIDisk( mm.getMajor() ) ||
785  block::MajorMinor::isIDEDisk( mm.getMajor() ) ||
786  block::MajorMinor::isVirtIODisk( mm ) ||
787  block::MajorMinor::isNVMeDisk( mm ) ||
788  block::MajorMinor::isMMCDisk( mm ) ||
789  block::MajorMinor::isBCacheDisk( mm ) ) {
790  detailDisk( mm, std::cout );
791  } else if ( block::MajorMinor::isMetaDisk( mm.getMajor() ) ) {
792  detailMetaDisk( mm, std::cout );
793  } else if ( mm.getClass() == block::DeviceMapper ) {
794  detailDeviceMapper( mm, std::cout );
795  } else if ( mm.getClass() == block::LVM ) {
796  detailLVM( mm, std::cout );
797  } else {
798  detailGenericBlockDevice( mm, std::cout );
799  }
800  }
801  } else {
802  std::cerr << "'" << options.device << "' is not a block device" << std::endl;
803  }
804  } else if ( options.opt_f ) {
805  if ( options.device != "" ) {
806  listFile( std::cout, options.device );
807  }
808  }
809  } else {
810  if ( options.opt_d ) {
811  listAllDisks( std::cout );
812  }
813  if ( options.opt_m ) {
814  listAllMetaDisks( std::cout );
815  }
816  if ( options.opt_l ) {

818  }

820  listAllDevices( std::cout );
821  }
822  }
823  } else printHelp();
824  } else printVersion();
825  }
