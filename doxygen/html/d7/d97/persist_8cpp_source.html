52  static void sqlite_ext_pow(sqlite3_context *context, int argc, sqlite3_value **argv) {
53  double r1 = 0.0;
54  double r2 = 0.0;
55  double val;
56 
57  assert( argc==2 );
58 
59  if( sqlite3_value_type(argv[0]) == SQLITE_NULL || sqlite3_value_type(argv[1]) == SQLITE_NULL ){
60  sqlite3_result_null(context);
61  } else {
62  r1 = sqlite3_value_double(argv[0]);
63  r2 = sqlite3_value_double(argv[1]);
64  errno = 0;
65  val = pow(r1,r2);
66  if (errno == 0) {
67  sqlite3_result_double(context, val);
68  } else {
69  sqlite3_result_error(context, strerror(errno), errno);
70  }
71  }
72  }

134  Database::Database( const std::string &filename, WaitHandler handler ) {
135  database_ = NULL;
136  int r = sqlite3_open_v2( filename.c_str(), &database_, SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE | SQLITE_OPEN_NOMUTEX, NULL );
137  if ( r != SQLITE_OK ) {
138  throw Oops( __FILE__, __LINE__, sqlite3_errmsg( database_ ) );
139  }
140  int eTextRep = SQLITE_UTF8;
141  #if SQLITE_VERSION_NUMBER > 3008003
142  eTextRep |= SQLITE_DETERMINISTIC;
143  #endif
144  r = sqlite3_create_function_v2( database_,
145  "power",
146  2,
147  eTextRep,
148  NULL,
149  sqlite_ext_pow,
150  NULL,
151  NULL,
152  NULL );
153  if ( r != SQLITE_OK ) {
154  throw Oops( __FILE__, __LINE__, sqlite3_errmsg( database_ ) );
155  }
156  r = sqlite3_create_function_v2( database_,
157  "ceil",
158  1,
159  eTextRep,
160  NULL,
161  sqlite_ext_ceil,
162  NULL,
163  NULL,
164  NULL );
165  if ( r != SQLITE_OK ) {
166  throw Oops( __FILE__, __LINE__, sqlite3_errmsg( database_ ) );
167  }
168  r = sqlite3_create_function_v2( database_,
169  "floor",
170  1,
171  eTextRep,
172  NULL,
173  sqlite_ext_floor,
174  NULL,
175  NULL,
176  NULL );
177  if ( r != SQLITE_OK ) {
178  throw Oops( __FILE__, __LINE__, sqlite3_errmsg( database_ ) );
179  }
180  r = sqlite3_create_function_v2( database_,
181  "log2",
182  1,
183  eTextRep,
184  NULL,
185  sqlite_ext_log2,
186  NULL,
187  NULL,
188  NULL );
189  if ( r != SQLITE_OK ) {
190  throw Oops( __FILE__, __LINE__, sqlite3_errmsg( database_ ) );
191  }
192 
193 
194  if ( handler ) sqlite3_busy_handler( database_, handler, database_ );
195  }
