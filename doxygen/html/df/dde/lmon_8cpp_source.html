83  bool verifyDatabase( persist::Database &db ) {
84  try {
85  persist::Query qempty(db);
86  qempty.prepare( "SELECT count(id) FROM snapshot" );
87  if ( qempty.step() ) {
88  long numsnaps = qempty.getLong(0);
89  if ( numsnaps < 1 ) {
90  std::cerr << "database stores 0 snapshots (lard running?)" << std::endl;
91  return false;
92  }
93  } else {
94  std::cerr << "invalid database" << std::endl;
95  return false;
96  }
97  }
98  catch ( const Oops& oops ) {
99  std::cerr << "invalid database" << std::endl;
100  return false;
101  }
102  return true;
103  }
