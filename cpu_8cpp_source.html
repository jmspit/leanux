75  void getCPUTopology( CPUTopology &topology ) {
76  topology.logical = 0;
77  topology.physical = 0;
78  topology.cores = 0;
79  DIR *d;
80  struct dirent *dir;
81  std::set<CoreId> cores;
82  std::set<unsigned long> physical;
83  d = opendir( (sysdevice::sysdevice_root + "/system/cpu").c_str() );
84  if ( d ) {
85  while ( (dir = readdir(d)) != NULL ) {
86  unsigned int cpunum = 0;
87  if ( sscanf( dir->d_name, "cpu%u", &cpunum ) == 1 ) {
88  topology.logical++;
89  CoreId cid;
90  cid.phy_id = util::fileReadUL( (std::string)sysdevice::sysdevice_root + "/system/cpu/" + dir->d_name + "/topology/physical_package_id" );
91  cid.core_id = util::fileReadUL( (std::string)sysdevice::sysdevice_root + "/system/cpu/" + dir->d_name + "/topology/core_id" );
92  cores.insert( cid );
93  physical.insert( cid.phy_id);
94  }
95  }
96  closedir( d );
97  } else throw Oops( __FILE__, __LINE__, "failed to read directory " + sysdevice::sysdevice_root + "/system/cpu" );
98  topology.physical = physical.size();
99  topology.cores = cores.size();
100  if ( topology.logical < 1 ||
101  topology.physical < 1 ||
102  topology.cores < 1 ||
103  topology.physical > topology.cores ||
104  topology.cores > topology.logical ) throw Oops( __FILE__, __LINE__, "the detected CPU topology is erratic" );
105 
106  };

122  std::string s;
123  int num_found = 0;

133  info.cpu_mhz = dv;
134  num_found++;
135  } else if ( sscanf( s.c_str(), "bogomips : %lf", &dv ) == 1 ) {

138  }
139  //PPC
140  else if ( strncmp( s.c_str(), "cpu : ", 6 ) == 0 ) {
141  info.model = s.substr(6);
142  num_found++;
143  } else if ( sscanf( s.c_str(), "clock : %lfMHz", &dv ) == 1 ) {
144  info.cpu_mhz = dv;

146  } else if ( sscanf( s.c_str(), "clock : %lf MHz", &dv ) == 1 ) {

157  void getCPUStats( CPUStatsMap &stats ) {
158  std::ifstream ifs( "/proc/stat" );
159  std::string s;
160  stats.clear();
161  while ( ifs.good() && !ifs.eof() ) {
162  std::getline( ifs, s );
163  if ( strncmp( s.c_str(), "cpu ", 4 ) != 0 ) { //skip total cpu line, as 'cpu 123' matches "cpu%u" in sscanf!
164  unsigned int cpunum = 0;
165  CPUStat stat;
166  memset( &stat, 0, sizeof(stat) );
167  unsigned long user = 0;
168  unsigned long nice = 0;
169  unsigned long system = 0;
170  unsigned long idle = 0;
171  unsigned long iowait = 0;
172  unsigned long irq = 0;
173  unsigned long softirq = 0;
174  unsigned long steal = 0;
175  unsigned long guest = 0;
176  unsigned long guest_nice = 0;
177  if ( sscanf( s.c_str(), "cpu%u %lu %lu %lu %lu %lu %lu %lu %lu %lu %lu",
178  &cpunum,
179  &user,
180  &nice,
181  &system,
182  &idle,
183  &iowait,
184  &irq,
185  &softirq,
186  &steal,
187  &guest,
188  &guest_nice
189  ) == 11 ) {
190  stat.user = (double)user / leanux::system::getUserHz();
191  stat.nice = (double)nice / leanux::system::getUserHz();
192  stat.system = (double)system / leanux::system::getUserHz();
193  stat.idle = (double)idle / leanux::system::getUserHz();
194  stat.iowait = (double)iowait / leanux::system::getUserHz();
195  stat.irq = (double)irq / leanux::system::getUserHz();
196  stat.softirq = (double)softirq / leanux::system::getUserHz();

199  stat.guest_nice = (double)guest_nice / leanux::system::getUserHz();
200  stats[cpunum] = stat;

229  bool deltaStats( const CPUStatsMap &earlier, const CPUStatsMap &later, CPUStatsMap &delta ) {
230  bool result = true;
231  delta.clear();
232  if ( earlier.size() != later.size() ) result = false; else {
233  for ( CPUStatsMap::const_iterator e = earlier.begin(); e != earlier.end(); ++e ) {
234  CPUStatsMap::const_iterator l = later.find( e->first );
235 
236  if ( l == later.end() ) {
237  result = false;
238  break;
239  } else {
240  if ( l->second.user >= e->second.user ) delta[e->first].user = l->second.user - e->second.user; else { result = false; break; };
241  if ( l->second.nice >= e->second.nice ) delta[e->first].nice = l->second.nice - e->second.nice; else { result = false; break; };
242  if ( l->second.system >= e->second.system ) delta[e->first].system = l->second.system - e->second.system; else { result = false; break; };
243  if ( l->second.idle >= e->second.idle ) delta[e->first].idle = l->second.idle - e->second.idle; else { result = false; break; };
244  if ( l->second.iowait >= e->second.iowait ) delta[e->first].iowait = l->second.iowait - e->second.iowait; else { result = false; break; };
245  if ( l->second.irq >= e->second.irq ) delta[e->first].irq = l->second.irq - e->second.irq; else { result = false; break; };
246  if ( l->second.softirq >= e->second.softirq ) delta[e->first].softirq = l->second.softirq - e->second.softirq; else { result = false; break; };
247  if ( l->second.steal >= e->second.steal ) delta[e->first].steal = l->second.steal - e->second.steal; else { result = false; break; };
248  if ( l->second.guest >= e->second.guest ) delta[e->first].guest = l->second.guest - e->second.guest; else { result = false; break; };
249  if ( l->second.guest_nice >= e->second.guest_nice ) delta[e->first].guest_nice = l->second.guest_nice - e->second.guest_nice; else { result = false; break; };
250  }
251  }
252  }
253  if (!result) delta.clear();
254  return result;
255  }

275  SchedInfo getSchedInfo() {
276  SchedInfo rq;
277  std::ifstream ifs( "/proc/stat" );
278  std::string s;
279  rq.running = 0;
280  rq.blocked = 0;
281  int match = 0;
282  while ( ifs.good() && !ifs.eof() && match < 4 ) {
283  getline( ifs, s );
284  if ( sscanf( s.c_str(), "procs_running %u", &rq.running ) == 1 ) match++;
285  else if ( sscanf( s.c_str(), "procs_blocked %u", &rq.blocked ) == 1 ) match++;
286  else if ( sscanf( s.c_str(), "processes %lu", &rq.processes ) == 1 ) match++;
287  else if ( sscanf( s.c_str(), "ctxt %lu", &rq.ctxt ) == 1 ) match++;
288  }
289  if ( match != 4 ) throw Oops( __FILE__, __LINE__, "/proc/stat parse failure" );
290  return rq;
291  }
